171  inline bool push(void *const data) {
172  unsigned long pw, seq;
173  element_t * node;
174  unsigned long bk = BACKOFF_MIN;
175  do {
176  pw = pwrite.load(std::memory_order_relaxed);
177  node = &buf[pw & mask];
178  seq = node->seq.load(std::memory_order_acquire);
179 
180  // load method
181  // Atomically loads and returns the current value of the atomic variable.
182  // Memory is affected according to the value of memory_order.
183

185  if (pwrite.compare_exchange_weak(pw, pw+1, std::memory_order_relaxed))
186  break;
187

190  bk <<= 1;

910  //
911  register long q, q1;
912  do {

914  if (q > q1) return false;

918  for(volatile unsigned i=0;i<bk;++i) ;
919  bk <<= 1;
920  bk &= BACKOFF_MAX;
921  } while(1);

924  if (pool[q].pop(data)) {
925  atomic_long_dec(&count);
926  return true;
927  }
928  return false;
929 
930 #else // MULTI_MPMC_RELAX_FIFO_ORDERING
931  register long q = atomic_long_inc_return(&dequeue) % pool.size();
932  bool r = pool[q].pop(data);
933  if (r) { atomic_long_dec(&count); return true;}
934  return false;

936  }
