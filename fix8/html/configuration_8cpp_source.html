67 int Configuration::process()
68 {
69  if (!exist(_xmlfile))
70  throw f8Exception("server config file not found", _xmlfile);
71  if (!_root)
72  throw f8Exception("could not create root xml entity");
73

75  throw f8Exception("could not locate server session in config file", _xmlfile);
76

80  load_map("fix8/ssl_context", _ssl_context);

83

107 size_t Configuration::get_addresses(const XmlElement *from, vector<Server>& target) const
108 {
109  string name;
110  const XmlElement *which;
111  if (from && from->GetAttr("server_group", name) && (which = find_server_group(name)))
112  {
113  XmlElement::XmlSet slist;
114  if (which->find("server_group/server", slist))
115  {
116  const Poco::Net::SocketAddress empty_addr;
117  for(XmlElement::XmlSet::const_iterator itr(slist.begin()); itr != slist.end(); ++itr)
118  {
119  string name;
120  Poco::Net::SocketAddress addr(get_address(*itr));
121  if ((*itr)->GetAttr("name", name) && addr != empty_addr && (*itr)->FindAttr("active", true))
122  target.push_back(Server(name, (*itr)->FindAttr("max_retries",
123  static_cast<int>(defaults::login_retries)), addr,
124  (*itr)->FindAttr("reset_sequence_numbers", false)));
125  }
126  }
127  return target.size();
128  }
129 
130  return 0;
131 }

134 Schedule Configuration::create_schedule(const XmlElement *which) const
135 {
136  Tickval start(get_time_field(which, "start_time", true));
137  if (!start.is_errorval())
138  {
139  const int utc_offset(which->FindAttr("utc_offset_mins", 0)); // utc_offset is in minutes
140  const unsigned duration(which->FindAttr("duration", 0));
141  Tickval end(get_time_field(which, "end_time", true));
142 
143  if (end.is_errorval())
144  {
145  if (duration) // duration is in minutes
146  end = start.get_ticks() + duration * Tickval::minute;
147  }
148  else
149  {
150  if (end <= start)
151  throw f8Exception("Schedule end time cannot be equal to or before session start time");
152  }
153  string daytmp;
154  const int start_day(which->GetAttr("start_day", daytmp) ? decode_dow(daytmp) : -1);
155  const int end_day(which->GetAttr("end_day", daytmp) ? decode_dow(daytmp) : start_day < 0 ? -1 : start_day);
156  return Schedule (start, end, duration, utc_offset, start_day, end_day);
157  }
158 
159  return Schedule();
160 }

189 Persister *Configuration::create_persister(const XmlElement *from, const SessionID *sid, bool flag) const
190 {
191  string name, type;
192  const XmlElement *which;
193  if (from && from->GetAttr("persist", name) && (which = find_persister(name)) && which->GetAttr("type", type))
194  {
195  if (type == "mem")
196  return new MemoryPersister;
197 
198  string dir("./"), db("persist_db");
199  which->GetAttr("dir", dir);
200  which->GetAttr("db", db) || which->GetAttr("session_prefix", db);
201 
202  if (sid)
203  db += ('.' + sid->get_senderCompID()() + '.' + sid->get_targetCompID()());
204  else if (which->FindAttr("use_session_id", false))
205  db += ('.' + get_sender_comp_id(from)() + '.' + get_target_comp_id(from)());
206 
207 #if defined HAVE_LIBMEMCACHED
208  if (type == "memcached")
209  {
210  string config_str;
211  if (which->GetAttr("config_string", config_str))
212  {
213  scoped_ptr<MemcachedPersister> result(new MemcachedPersister);
214  if (result->initialise(config_str, db, flag))
215  return result.release();
216  }
217  else
218  throw f8Exception("memcached:config_string attribute must be given when using memcached.");
219  }
220  else
221 #endif
222 #if defined HAVE_LIBHIREDIS
223  if (type == "redis")
224  {
225  string host_str;
226  if (which->GetAttr("host", host_str))
227  {
228  scoped_ptr<HiredisPersister> result(new HiredisPersister);
229  if (result->initialise(host_str, which->FindAttr("port", 6379),
230  which->FindAttr("connect_timeout", static_cast<unsigned>(defaults::connect_timeout)), db, flag))
231  return result.release();
232  }
233  else
234  throw f8Exception("redis:host attribute must be given when using redis.");
235  }
236  else
237 #endif
238 #if defined HAVE_BDB
239  if (type == "bdb")
240  {
241  scoped_ptr<BDBPersister> result(new BDBPersister);
242  if (result->initialise(dir, db, flag))
243  return result.release();
244  }
245  else
246 #endif
247  if (type == "file")
248  {
249  scoped_ptr<FilePersister> result(new FilePersister(which->FindAttr("rotation", 0)));
250  if (result->initialise(dir, db, flag))
251  return result.release();
252  }
253  }
254 
255  return 0;
256 }

259 Logger *Configuration::create_logger(const XmlElement *from, const Logtype ltype, const SessionID *sid) const
260 {
261  string name;
262  if (from && from->GetAttr(ltype == session_log ? "session_log" : "protocol_log", name))
263  {
264  const XmlElement *which(find_logger(name));
265  if (which)
266  {
267  string type;
268  if (which->GetAttr("type", type)
269  && ((type % "session" && ltype == session_log) || (type % "protocol" && ltype == protocol_log)))
270  {
271  string logname("logname_not_set.log");
272  which->FindAttrRef("filename", logname);
273  trim(logname);
274 
275  if (logname[0] == '|')
276 #ifndef HAVE_POPEN
277  throw f8Exception("popen not supported on your platform");
278 #endif
279  return new PipeLogger(logname, get_logflags(which));
280 
281  RegMatch match;
282  if (_ipexp.SearchString(match, logname, 3) == 3)
283  {
284  f8String ip, port;
285  _ipexp.SubExpr(match, logname, ip, 0, 1);
286  _ipexp.SubExpr(match, logname, port, 0, 2);
287  BCLogger *bcl(new BCLogger(ip, get_value<unsigned>(port), get_logflags(which)));
288  if (*bcl)
289  return bcl;
290  }
291 
292  get_logname(which, logname, sid); // only applies to file loggers
293  return new FileLogger(logname, get_logflags(which), get_logfile_rotation(which));
294  }
295  }
296  }
297 
298  return 0;
299 }

359  target._certificate_file = which->FindAttrRef("ceritificte_file", empty);

361  target._verification_depth = which->FindAttr("verification_depth", static_cast<int>(defaults::verification_depth));
362  target._load_default_cas = which->FindAttr("load_default_cas", false);
363  target._cipher_list = which->FindAttrRef("cipher_list", cipher);
364  name = which->FindAttrRef("verification_mode", relaxed);
365  if (name == "none")
366  target._verification_mode = Poco::Net::Context::VERIFY_NONE;
367  else if (name == "relaxed")
368  target._verification_mode = Poco::Net::Context::VERIFY_RELAXED;
369  else if (name == "strict")
370  target._verification_mode = Poco::Net::Context::VERIFY_STRICT;
371  else if (name == "once")
372  target._verification_mode = Poco::Net::Context::VERIFY_ONCE;
373  else
