64 int Configuration::process()
65 {
66  if (!exist(_xmlfile))
67  throw f8Exception("server config file not found", _xmlfile);
68  if (!_root)
69  throw f8Exception("could not create root xml entity");
70 
71  if (!load_map("fix8/session", _sessions, true))
72  throw f8Exception("could not locate server session in config file", _xmlfile);
73 
74  load_map("fix8/persist", _persisters);
75  load_map("fix8/log", _loggers);
76 
77  return _sessions.size();
78 }

138 Logger *Configuration::create_logger(const XmlElement *from, const Logtype ltype) const
139 {
140  string name;
141  if (from && from->GetAttr(ltype == session_log ? "session_log" : "protocol_log", name))
142  {
143  const XmlElement *which(find_logger(name));
144  if (which)
145  {
146  string type;
147  if (which->GetAttr("type", type)
148  && ((type % "session" && ltype == session_log) || (type % "protocol" && ltype == protocol_log)))
149  {
150  string logname("logname_not_set.log");
151  trim(get_logname(which, logname));
152 
153  if (logname[0] == '|' || logname[0] == '!')
154  return new PipeLogger(logname, get_logflags(which));
155 
156  RegMatch match;
157  if (_ipexp.SearchString(match, logname, 3) == 3)
158  {
159  f8String ip, port;
160  _ipexp.SubExpr(match, logname, ip, 0, 1);
161  _ipexp.SubExpr(match, logname, port, 0, 2);
162  BCLogger *bcl(new BCLogger(ip, GetValue<unsigned>(port), get_logflags(which)));
163  if (*bcl)
164  return bcl;
165  }
166 
167  return new FileLogger(logname, get_logflags(which), get_logfile_rotation(which));
168  }
169  }
170  }
171 
172  return 0;
173 }

176 Logger::LogFlags Configuration::get_logflags(const XmlElement *from) const
177 {
178  Logger::LogFlags flags;
179  string flags_str;
180  if (from && from->GetAttr("flags", flags_str))
181  {
182  istringstream istr(flags_str);
183  for(char extr[32]; !istr.get(extr, sizeof(extr), '|').eof() || extr[0]; istr.ignore(1))
184  {
185  string result(extr);
186  flags.set(Logger::num_flags, Logger::_bit_names, trim(result));
187  }
188  }
189 
190  return flags;
191 }
