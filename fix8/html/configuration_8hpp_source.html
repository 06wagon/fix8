93  unsigned load_map(const std::string& tag, ConfigMap& map_name, const bool is_session=false)
94  {
95  XmlElement::XmlSet slist;
96  if (_root->find(tag, slist))
97  {
98  for(XmlElement::XmlSet::const_iterator itr(slist.begin()); itr != slist.end(); ++itr)
99  {
100  std::string name;
101  if ((*itr)->GetAttr("name", name) && is_session ? (*itr)->FindAttr("active", false) : true)
102  {
103  map_name.insert(ConfigMap::value_type(name, *itr));
104  if (is_session)
105  _allsessions.push_back(*itr);
106  }
107  }
108  }
109 
110  return map_name.size();
111  }
