119  unsigned load_map(const std::string& tag, ConfigMap& map_name, const bool is_session=false)
120  {
121  XmlElement::XmlSet slist;
122  if (_root->find(tag, slist))
123  {
124  for(XmlElement::XmlSet::const_iterator itr(slist.begin()); itr != slist.end(); ++itr)
125  {
126  std::string name;
127  if ((*itr)->GetAttr("name", name) && is_session ? (*itr)->FindAttr("active", false) : true)
128  {
129  map_name.insert(ConfigMap::value_type(name, *itr));
130  if (is_session)
131  _allsessions.push_back(*itr);
132  }
133  }
134  }
135 
136  return map_name.size();
137  }
