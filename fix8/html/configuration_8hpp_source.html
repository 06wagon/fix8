181  unsigned load_map(const std::string& tag, ConfigMap& map_name, const bool is_session=false)
182  {
183  XmlElement::XmlSet slist;
184  if (_root->find(tag, slist))
185  {
186  for(XmlElement::XmlSet::const_iterator itr(slist.begin()); itr != slist.end(); ++itr)
187  {
188  std::string name;
189  if ((*itr)->GetAttr("name", name) && is_session ? (*itr)->FindAttr("active", false) : true)
190  {
191  map_name.insert(ConfigMap::value_type(name, *itr));
192  if (is_session)
193  _allsessions.push_back(*itr);
194  }
195  }
196  }
197 
198  return map_name.size();
199  }
