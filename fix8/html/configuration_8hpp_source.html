183  unsigned load_map(const std::string& tag, ConfigMap& map_name, const bool is_session=false)
184  {
185  XmlElement::XmlSet slist;
186  if (_root->find(tag, slist))
187  {
188  for(const auto *pp : slist)
189  {
190  std::string name;
191  if (pp->GetAttr("name", name) && is_session ? pp->FindAttr("active", false) : true)
192  {
193  map_name.insert({name, pp});
194  if (is_session)
195  _allsessions.push_back(pp);
196  }
197  }
198  }
199 
200  return static_cast<unsigned>(map_name.size());
201  }
