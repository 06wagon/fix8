151 bool FIXReader::read(f8String& to) // read a complete FIX message
152 {
153  char msg_buf[_max_msg_len] = {};
154  int result(sockRead(msg_buf, _bg_sz));
155 
156  if (result == static_cast<int>(_bg_sz))
157  {
158  char bt;
159  size_t offs(_bg_sz);
160  do // get the last chrs of bodylength and ^A
161  {
162  if (sockRead(&bt, 1) != 1)
163  return false;
164  if (!isdigit(bt) && bt != default_field_separator)
165  throw IllegalMessage(msg_buf);
166  msg_buf[offs++] = bt;
167  }
168  while (bt != default_field_separator && offs < _max_msg_len);
169  to.assign(msg_buf, offs);
170 
171  f8String tag, bgstr, len;
172  unsigned result;
173  if ((result = MessageBase::extract_element(to.data(), to.size(), tag, bgstr)))
174  {
175  if (tag != "8")
176  throw IllegalMessage(to);
177 
178  if (bgstr != _session.get_ctx()._beginStr) // invalid FIX version
179  throw InvalidVersion(bgstr);
180 
181  if ((result = MessageBase::extract_element(to.data() + result, to.size() - result, tag, len)))
182  {
183  if (tag != "9")
184  throw IllegalMessage(to);
185 
186  const unsigned mlen(fast_atoi<unsigned>(len.c_str()));
187  if (mlen == 0 || mlen > _max_msg_len - _bg_sz - _chksum_sz) // invalid msglen
188  throw InvalidBodyLength(mlen);
189 
190  // read the body
191  if ((result = sockRead(msg_buf, mlen) != static_cast<int>(mlen)))
192  return false;
193 
194  // read the checksum
195  if ((result = sockRead(msg_buf + mlen, _chksum_sz) != static_cast<int>(_chksum_sz)))
196  return false;
197 
198  to.append(msg_buf, mlen + _chksum_sz);
199  _session.update_received();
200  return true;
201  }
202  }
203 
204  throw IllegalMessage(to);
205  }
206 
207  return false;
208 }

215 
216  for (; !_session.is_shutdown();)
