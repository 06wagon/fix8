121  int sockRead(char *where, const size_t sz)
122  {
123  unsigned remaining(sz), rddone(0);
124 
125  while (remaining > 0)
126  {
127  const int rdSz(_sock->receiveBytes(where + rddone, remaining));
128  if (rdSz <= 0)
129  {
130  switch(errno)
131  {
132  case EAGAIN:
133 #if defined EWOULDBLOCK && EAGAIN != EWOULDBLOCK
134  case EWOULDBLOCK:
135 #endif

137  default:
138  throw PeerResetConnection("sockRead: connection gone");
