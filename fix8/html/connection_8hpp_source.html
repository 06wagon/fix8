160  const size_t available_in_buffer(static_cast<size_t>(_read_buffer_wptr - _read_buffer_rptr));
161  if (available_in_buffer < sz)
162  realSockRead(sz - available_in_buffer, _max_msg_len);
163  sz = std::min((size_t)(_read_buffer_wptr-_read_buffer_rptr), sz);
164  memcpy(where, _read_buffer_rptr, sz);
165  _read_buffer_rptr += sz;
166  const size_t shift(_max_msg_len);
167  if (static_cast<size_t>(_read_buffer_rptr - _read_buffer) >= shift)

181  int realSockRead(size_t sz, size_t maxsz)
182  {
183  const size_t max_sz(_read_buffer + sizeof(_read_buffer) - _read_buffer_wptr);
184  int maxremaining(std::min(maxsz, max_sz)), remaining(std::min(sz, max_sz));
185  char *ptr(_read_buffer_wptr), *eptr(_read_buffer + sizeof(_read_buffer));
186 
187  int rdsz(0);
188  while (remaining > 0 && _read_buffer_wptr < eptr)
189  {
190  rdsz = _sock->receiveBytes(_read_buffer_wptr, maxremaining);
191  if (rdsz <= 0)
192  {
193  if (errno == EAGAIN
194 #if defined EWOULDBLOCK && EAGAIN != EWOULDBLOCK
195  || errno == EWOULDBLOCK
196 #endif
197  )
198  continue;
199  throw PeerResetConnection("sockRead: connection gone");
200  }
201  _read_buffer_wptr += rdsz;
202  remaining -= rdsz;
203  maxremaining -= rdsz;
204  }

363  size_t write_batch(const std::vector<Message *>& msgs, bool destroy)
364  {
365  if (msgs.empty())
366  return 0;
367  if (msgs.size() == 1)
368  return write(msgs.front(), destroy) ? 1 : 0;
369  size_t result(0);
370  f8_scoped_spin_lock guard(_con_spl);
371  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()), litr(eitr-1); itr != eitr; ++itr)
372  {
373  Message* msg = *itr;
374  msg->set_end_of_batch(itr == litr);
375  if (_pmodel == pm_pipeline) // pipeline mode ignores destroy flag
376  {
377  _msg_queue.try_push(msg);
378  ++result;
379  continue;
380  }
381  if (_session.send_process(msg))
382  ++result;
383  }
384  if (destroy && _pmodel != pm_pipeline)
385  {
386  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()); itr != eitr; ++itr)
387  {
388  std::unique_ptr<Message> smsg(*itr);
389  }
390  }
392  return result;
393  }
396  int join() { return _pmodel == pm_pipeline ? AsyncSocket<Message *>::join() : -1; }
