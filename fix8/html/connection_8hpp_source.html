358  size_t write_batch(const std::vector<Message *>& msgs, bool destroy)
359  {
360  if (msgs.empty())
361  return 0;
362  if (msgs.size() == 1)
363  return write(msgs.front(), destroy) ? 1 : 0;
364  size_t result(0);
365  f8_scoped_spin_lock guard(_con_spl);
366  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()), litr(eitr-1); itr != eitr; ++itr)
367  {
368  Message* msg = *itr;
369  msg->set_end_of_batch(itr == litr);
370  if (_pmodel == pm_pipeline) // pipeline mode ignores destroy flag
371  {
372  _msg_queue.try_push(msg);
373  ++result;
374  continue;
375  }
376  if (_session.send_process(msg))
377  ++result;
378  }
379  if (destroy && _pmodel != pm_pipeline)
380  {
381  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()); itr != eitr; ++itr)
382  {
383  std::unique_ptr<Message> smsg(*itr);
384  }
385  }
387  return result;
388  }
391  int join() { return _pmodel == pm_pipeline ? AsyncSocket<Message *>::join() : -1; }
