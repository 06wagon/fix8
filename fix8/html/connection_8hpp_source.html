186  const size_t max_sz(_read_buffer + sizeof(_read_buffer) - _read_buffer_wptr);

366  size_t write_batch(const std::vector<Message *>& msgs, bool destroy)
367  {
368  if (msgs.empty())
369  return 0;
370  if (msgs.size() == 1)
371  return write(msgs.front(), destroy) ? 1 : 0;
372  size_t result(0);
373  f8_scoped_spin_lock guard(_con_spl);
374  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()), litr(eitr-1); itr != eitr; ++itr)
375  {
376  Message* msg = *itr;
377  msg->set_end_of_batch(itr == litr);
378  if (_pmodel == pm_pipeline) // pipeline mode ignores destroy flag
379  {
380  _msg_queue.try_push(msg);
381  ++result;
382  continue;
383  }
384  if (_session.send_process(msg))
385  ++result;
386  }
387  if (destroy && _pmodel != pm_pipeline)
388  {
389  for (std::vector<Message *>::const_iterator itr(msgs.begin()), eitr(msgs.end()); itr != eitr; ++itr)
390  {
391  std::unique_ptr<Message> smsg(*itr);
392  }
393  }
395  return result;
396  }
399  int join() { return _pmodel == pm_pipeline ? AsyncSocket<Message *>::join() : -1; }
400

410  }
