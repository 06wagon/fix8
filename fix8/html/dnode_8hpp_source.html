250  virtual inline bool pop(void ** ptr) {
251  if (skipdnode || P) return ff_node::pop(ptr);
252 
253  // gets the peers involved in one single communication
254  const int sendingPeers=com.getToWait();
255 #ifdef _WIN32
256  svector<msg_t*> ** v = (svector<msg_t*> **) malloc(sendingPeers*sizeof(svector<msg_t*> *));
257 #else // C99
258  svector<msg_t*>* v[sendingPeers];
259 #endif
260  for(int i=0;i<sendingPeers;++i) {
261  msg_t hdr;
262  int sender=-1;
263  if (!com.gethdr(hdr, sender)) {
264  error("dnode:pop: ERROR: receiving header from peer");
265  // free (v); // Win
266  return false;
267  }
268  if (isEos(static_cast<char *>(hdr.getData()))) {
269  //std::cerr << "DNODE gets DEOS\n";
270  if (++neos==com.getDescriptor()->getPeers()) {
271  com.done();
272  *ptr = (void*)FF_EOS;
273  neos=0;
274  // free (v) // Win
275  return true;
276  }
277  if (sendingPeers==1) i=-1; // reset for index
278  continue;
279  }
280  uint32_t len = *static_cast<uint32_t*>(hdr.getData());
281  register int ventry = (sendingPeers==1)?0:sender;
282  prepare(v[ventry], len, sender);
283  assert(v[ventry]->size() == len);
284 
285  for(size_t j=0;j<len;++j)
286  if (!com.get(*(v[ventry]->operator[](j)),sender)) {
287  error("dnode:pop: ERROR: receiving data from peer");
288  // free (v) // Win32
289  return false;
290  }
291  }
292  com.done();
293 
294  unmarshalling(v, sendingPeers, *ptr);
295  // free (v) // Win32
296  return true;
297  }
