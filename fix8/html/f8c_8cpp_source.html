397 int load_fields(XmlElement& xf, FieldSpecMap& fspec)
398 {
399  int fieldsLoaded(0);
400 
401  XmlElement::XmlSet flist;
402  if (!xf.find("fix/fields/field", flist))
403  {
404  cerr << "error: No fields found in " << shortName << endl;
405  return 0;
406  }
407 
408  for(XmlElement::XmlSet::const_iterator itr(flist.begin()); itr != flist.end(); ++itr)
409  {
410  string number, name, type;
411  if ((*itr)->GetAttr("number", number) && (*itr)->GetAttr("name", name) && (*itr)->GetAttr("type", type))
412  {
413  trim(number);
414  trim(name);
415  trim(type);
416  InPlaceStrToUpper(type);
417  FieldTrait::FieldType ft(FieldSpec::_baseTypeMap.find_value(type));
418  pair<FieldSpecMap::iterator, bool> result;
419  if (ft != FieldTrait::ft_untyped)
420  result = fspec.insert(FieldSpecMap::value_type(get_value<unsigned>(number), FieldSpec(name, ft)));
421  else
422  {
423  cerr << shortName << ':' << recover_line(**itr) << ": warning: Unknown field type: " << type << " in " << name << endl;
424  ++glob_warnings;
425  continue;
426  }
427 
428  (*itr)->GetAttr("description", result.first->second._description);
429  (*itr)->GetAttr("comment", result.first->second._comment);
430 
431  ++fieldsLoaded;
432 
433  XmlElement::XmlSet realmlist;
434  if ((*itr)->find("field/value", realmlist))
435  {
436  for(XmlElement::XmlSet::const_iterator ditr(realmlist.begin()); ditr != realmlist.end(); ++ditr)
437  {
438  string enum_str, description;
439  if ((*ditr)->GetAttr("enum", enum_str))
440  {
441  if (!(*ditr)->GetAttr("description", description) || description.empty())
442  description = enum_str; // use enum if no description supplied
443 
444  if (!result.first->second._dvals)
445  result.first->second._dvals = new RealmMap;
446  string lower, upper;
447  bool isRange((*ditr)->GetAttr("range", lower) && (lower == "lower" || upper == "upper"));
448  RealmObject *realmval(RealmObject::create(enum_str, ft, isRange));
449  if (isRange)
450  result.first->second._dtype = RealmBase::dt_range;
451  if (realmval)
452  result.first->second._dvals->insert(RealmMap::value_type(realmval, description));
453  }
454  else
455  {
456  cerr << shortName << ':' << recover_line(**itr) << ": error: Value element missing required attributes." << endl;
457  ++glob_errors;
458  }
459  }
460  }
461  }
462  else
463  {
464  cerr << shortName << ':' << recover_line(**itr) << ": error: Field element missing required attributes" << endl;
465  ++glob_errors;
466  }
467  }
468 
469  return fieldsLoaded;
470 }

473 int load_messages(XmlElement& xf, MessageSpecMap& mspec, const FieldToNumMap& ftonSpec,
474  FieldSpecMap& fspec, Components& compon, GroupMap& globmap)
475 {
476  int msgssLoaded(0), grpsparsed(0);
477 
478  XmlElement::XmlSet mlist;
479  if (!xf.find("fix/messages/message", mlist))
480  {
481  cerr << "error: No messages found in " << shortName << endl;
482  ++glob_errors;
483  return 0;
484  }
485 
486  if (!xf.find("fix/header", mlist))
487  {
488  cerr << "error: No header element found in " << shortName << endl;
489  ++glob_errors;
490  return 0;
491  }
492 
493  if (!xf.find("fix/trailer", mlist))
494  {
495  cerr << "error: No trailer element found in " << shortName << endl;
496  ++glob_errors;
497  return 0;
498  }
499 
500  // lookup msgtype realm - all messages must have corresponding entry here
501  FieldSpecMap::const_iterator fsitr(fspec.find(35)); // always 35
502  if (fsitr == fspec.end() || !fsitr->second._dvals)
503  {
504  cerr << "error: Could not locate MsgType realm defintions in " << shortName
505  << ". See FAQ for more details." << endl;
506  ++glob_errors;
507  return 0;
508  }
509 
510  for(XmlElement::XmlSet::const_iterator itr(mlist.begin()); itr != mlist.end(); ++itr)
511  {
512  string msgcat, name, msgtype, elname;
513  if ((*itr)->GetTag() == "header")
514  msgtype = name = elname = "header";
515  else if ((*itr)->GetTag() == "trailer")
516  msgtype = name = elname = "trailer";
517  else if ((*itr)->GetAttr("msgtype", msgtype) && (*itr)->GetAttr("name", name) && (*itr)->GetAttr("msgcat", msgcat))
518  {
519  StringRealm srealm(msgtype, false);
520  RealmMap::const_iterator ditr(fsitr->second._dvals->find(&srealm));
521  if (ditr == fsitr->second._dvals->end())
522  {
523  cerr << shortName << ':' << recover_line(**itr) << ": error: Message "
524  << name << " does not have corrresponding entry in MsgType field realm. See FAQ for more details." << endl;
525  ++glob_errors;
526  continue;
527  }
528 
529  elname = "message";
530  }
531  else
532  {
533  cerr << shortName << ':' << recover_line(**itr) << ": error: Message element missing required attributes" << endl;
534  ++glob_errors;
535  continue;
536  }
537 
538  pair<MessageSpecMap::iterator, bool> result(
539  mspec.insert(MessageSpecMap::value_type(msgtype, MessageSpec(name, msgcat % "admin"))));
540  if (!result.second)
541  {
542  cerr << shortName << ':' << recover_line(**itr) << ": error: Could not add message " << name << " (" << msgtype << ")" << endl;
543  ++glob_errors;
544  continue;
545  }
546 
547  string elpart;
548  XmlElement::XmlSet grplist;
549  if ((*itr)->find(mkel(elname, "group", elpart), grplist))
550  grpsparsed += parse_groups(result.first->second, itr, name, ftonSpec, fspec, grplist, compon, globmap);
551 
552  (*itr)->GetAttr("comment", result.first->second._comment);
553 
554  process_message_fields(mkel(elname, "field", elpart), *itr, result.first->second._fields, ftonSpec, fspec, compon);
555 
556  ++msgssLoaded;
557  }
558 
559  if (verbose)
560  cout << grpsparsed << " groups processed" << endl;
561 
562  return msgssLoaded;
563 }

566 unsigned parse_groups(MessageSpec& ritr, XmlElement::XmlSet::const_iterator& itr, const string& name,
567  const FieldToNumMap& ftonSpec, FieldSpecMap& fspec, XmlElement::XmlSet& grplist,
568  const Components& compon, GroupMap& globmap)
569 {
570  unsigned result(0);
571 
572  for(XmlElement::XmlSet::const_iterator gitr(grplist.begin()); gitr != grplist.end(); ++gitr)
573  {
574  string gname, required;
575  if ((*gitr)->GetAttr("name", gname) && (*gitr)->GetAttr("required", required))
576  {
577  string compname;
578  unsigned compidx((*gitr)->GetAttr("component", compname) ? lookup_component(compon, compname) : 0);
579 
580  // add group FieldTrait
581  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(gname));
582  FieldSpecMap::const_iterator fs_itr;
583  if (ftonItr != ftonSpec.end() && (fs_itr = fspec.find(ftonItr->second)) != fspec.end())
584  {
585  if (!ritr._fields.add(FieldTrait(fs_itr->first, FieldTrait::ft_int, (*gitr)->GetSubIdx(),
586  required == "Y", true, compidx)))
587  {
588  cerr << "error: Could not add group trait object " << gname << endl;
589  ++glob_errors;
590  }
591  else
592  {
593  fs_itr->second._used = true; // we always assume group count fields are used
594  pair<GroupMap::iterator, bool> gresult(
595  ritr._groups.insert(GroupMap::value_type(fs_itr->first, MessageSpec(gname))));
596  if (gresult.second)
597  {
598  process_message_fields("group/field", *gitr, gresult.first->second._fields, ftonSpec, fspec, compon);
599  XmlElement::XmlSet comlist;
600  ++result;
601  if ((*gitr)->find("group/group", comlist))
602  result += parse_groups(gresult.first->second, gitr, gname, ftonSpec, fspec, comlist, compon, globmap);
603  globmap.insert(*gresult.first);
604  GroupMap::iterator ritr(globmap.find(fs_itr->first));
605  if (ritr != globmap.end())
606  ritr->second.group_refcnt++;
607  }
608  else
609  {
610  cerr << "error: Could not add group map " << fs_itr->first << endl;
611  ++glob_errors;
612  }
613  }
614  }
615  else
616  {
617  cerr << shortName << ':' << recover_line(**itr)
618  << ": error: Could not locate group Field " << gname << " from known field types in " << shortName << endl;
619  ++glob_errors;
620  continue;
621  }
622  }
623  else
624  {
625  cerr << shortName << ':' << recover_line(**itr) << ": error: Group element missing required attributes" << endl;
626  ++glob_errors;
627  }
628  }
629 
630  return result;
631 }
