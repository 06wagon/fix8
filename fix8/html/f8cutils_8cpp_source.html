00146       cerr << "Unsupported FIX version " << ctxt._version << " from fix header in " << shortName << endl;

00152    ctxt._systemns = ostr.str();

00154       ctxt._fixns = ctxt._systemns;
00155    ctxt._clname = prefix;
00156

00244             {
00245                cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
00246                ++glob_errors;
00247                continue;
00248             }
00249 
00250             // add FieldTrait
00251             if (!fts.add(FieldTrait(fs_itr->first, fs_itr->second._ftype, (*fitr)->GetSubIdx(), required == "Y", false, subpos)))
00252                cerr << shortName << ':' << recover_line(**fitr) << ": error: Could not add trait object " << fname << endl;
00253             else
00254             {
00255                process_special_traits(fs_itr->first, fts);
00256                ++processed;
00257                fs_itr->second.set_used();
00258             }
00259          }
00260          else
00261          {
00262             cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
00263             ++glob_errors;
