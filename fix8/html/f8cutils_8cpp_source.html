131 int load_fix_version (XmlElement& xf, Ctxt& ctxt)
132 {
133  const XmlElement *fix(xf.find("fix"));
134  if (!fix)
135  {
136  cerr << "No fix header element found in " << shortName << endl;
137  return -1;
138  }
139 
140  string major, minor, revision("0"), type("FIX");
141 
142  if (!fix->GetAttr("major", major) || !fix->GetAttr("minor", minor))
143  {
144  cerr << "Missing required attributes (major/minor) from fix header in " << shortName << endl;
145  return -1;
146  }
147 
148  if (!fix->GetAttr("revision", revision))
149  fix->GetAttr("servicepack", revision);
150  fix->GetAttr("type", type);
151 
152  // fix version: <Major:1><Minor:1><Revision:2> eg. 4.2r10 is 4210
153  ctxt._version = GetValue<int>(major) * 1000 + GetValue<int>(minor) * 100 + GetValue<int>(revision);
154  if (type == "FIX" && ctxt._version < 4000)
155  {
156  cerr << "Unsupported FIX version " << ctxt._version << " from fix header in " << shortName << endl;
157  return -1;
158  }
159 
160  ostringstream ostr;
161  ostr << type << ctxt._version;
162  ctxt._systemns = ostr.str();
163  if (ctxt._fixns.empty())
164  ctxt._fixns = ctxt._systemns;
165  ctxt._clname = prefix;
166 
167  ostr.str("");
168  ostr << type << '.' << major << '.' << minor;
169  ctxt._beginstr = ostr.str();
170 
171  return 0;
172 }

191 void process_value_enums(FieldSpecMap::const_iterator itr, ostream& ost_hpp, ostream& ost_cpp)
192 {
193  string typestr("const ");
194  if (FieldTrait::is_int(itr->second._ftype))
195  typestr += "int ";
196  else if (FieldTrait::is_char(itr->second._ftype))
197  typestr += "char ";
198  else if (FieldTrait::is_float(itr->second._ftype))
199  typestr += "double ";
200  else if (FieldTrait::is_string(itr->second._ftype))
201  typestr += "f8String ";
202  else
203  return;
204 
205  ost_cpp << typestr << itr->second._name << "_realm[] = " << endl << spacer << "{ ";
206  unsigned cnt(0);
207  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
208  {
209  if (cnt)
210  ost_cpp << ", ";
211  ost_cpp << *ditr->first;
212  string transdesc(ditr->second);
213  // replace any illegal c++ identifier characters
214  ost_hpp << typestr << itr->second._name << '_' << InPlaceReplaceInSet(ident_set, transdesc, '_');
215  if (ditr->first->is_range())
216  {
217  if (cnt == 0)
218  ost_hpp << "_lower";
219  else if (cnt == 1)
220  ost_hpp << "_upper";
221  }
222  ost_hpp << '(' << *ditr->first << ");" << endl;
223  ++cnt;
224  }
225  ost_hpp << "const size_t " << itr->second._name << "_realm_els(" << itr->second._dvals->size() << ");" << endl;
226  ost_cpp << " };" << endl;
227 
228  ost_cpp << "const char *" << itr->second._name << "_descriptions[] = " << endl << spacer << "{ ";
229  cnt = 0;
230  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
231  {
232  if (cnt)
233  ost_cpp << ", ";
234  ost_cpp << '"' << ditr->second << '"';
235  ++cnt;
236  }
237  ost_cpp << " };" << endl;
238 }

241 int process_message_fields(const std::string& where, const XmlElement *xt, FieldTraits& fts, const FieldToNumMap& ftonSpec,
242  FieldSpecMap& fspec, const unsigned subpos)
243 {
244  unsigned processed(0);
245  XmlElement::XmlSet flist;
246  if (xt->find(where, flist))
247  {
248  for(XmlElement::XmlSet::const_iterator fitr(flist.begin()); fitr != flist.end(); ++fitr)
249  {
250  string fname, required;
251  if ((*fitr)->GetAttr("name", fname) && (*fitr)->GetAttr("required", required))
252  {
253  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(fname));
254  FieldSpecMap::iterator fs_itr;
255  if (ftonItr == ftonSpec.end() || (fs_itr = fspec.find(ftonItr->second)) == fspec.end())
256  {
257  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
258  ++glob_errors;
259  continue;
260  }
261 
262  // add FieldTrait
263  if (!fts.add(FieldTrait(fs_itr->first, fs_itr->second._ftype, (*fitr)->GetSubIdx(), required == "Y", false, subpos)))
264  cerr << shortName << ':' << recover_line(**fitr) << ": error: Could not add trait object " << fname << endl;
265  else
266  {
267  process_special_traits(fs_itr->first, fts);
268  ++processed;
269  fs_itr->second.set_used();
270  }
271  }
272  else
273  {
274  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
275  ++glob_errors;
276  }
277  }
278  }
279 
280  return processed;
281 }

290 void process_ordering(MessageSpecMap& mspec)
291 {
292  for (MessageSpecMap::const_iterator mitr(mspec.begin()); mitr != mspec.end(); ++mitr)
293  {
294  FieldTraitOrder mo;
295  for (Presence::const_iterator flitr(mitr->second._fields.get_presence().begin());
296  flitr != mitr->second._fields.get_presence().end(); ++flitr)
297  mo.insert(FieldTraitOrder::value_type(&*flitr));
298 
299  unsigned cnt(0);
300  for (FieldTraitOrder::iterator fto(mo.begin()); fto != mo.end(); ++fto)
301  (*fto)->_pos = ++cnt;
302 
303  process_group_ordering(mitr->second);
304  }
305 }

309 {

316 
317  unsigned gcnt(0);
318  for (FieldTraitOrder::iterator fto(go.begin()); fto != go.end(); ++fto)
319  (*fto)->_pos = ++gcnt;
320 
321  if (!gitr->second._groups.empty())
322  process_group_ordering(gitr->second);
323  }

355 RealmObject *RealmObject::create(const string& from, FieldTrait::FieldType ftype, bool isRange)
356 {
357  if (FieldTrait::is_int(ftype))
358  return new TypedRealm<int>(GetValue<int>(from), isRange);
359  if (FieldTrait::is_char(ftype))
360  return new CharRealm(from[0], isRange);
361  if (FieldTrait::is_float(ftype))
362  return new TypedRealm<double>(GetValue<double>(from), isRange);
363  if (FieldTrait::is_string(ftype))
364  return new StringRealm(from, isRange);
365  return 0;
