130 int load_fix_version (XmlElement& xf, Ctxt& ctxt)
131 {
132  const XmlElement *fix(xf.find("fix"));
133  if (!fix)
134  {
135  cerr << "No fix header element found in " << shortName << endl;
136  return -1;
137  }
138 
139  string major, minor, revision("0"), type("FIX");
140 
141  if (!fix->GetAttr("major", major) || !fix->GetAttr("minor", minor))
142  {
143  cerr << "Missing required attributes (major/minor) from fix header in " << shortName << endl;
144  return -1;
145  }
146 
147  if (!fix->GetAttr("revision", revision))
148  fix->GetAttr("servicepack", revision);
149  fix->GetAttr("type", type);
150 
151  // fix version: <Major:1><Minor:1><Revision:2> eg. 4.2r10 is 4210
152  ctxt._version = get_value<int>(major) * 1000 + get_value<int>(minor) * 100 + get_value<int>(revision);
153  if (type == "FIX" && ctxt._version < 4000)
154  {
155  cerr << "Unsupported FIX version " << ctxt._version << " from fix header in " << shortName << endl;
156  return -1;
157  }
158 
159  ostringstream ostr;
160  ostr << type << ctxt._version;
161  ctxt._systemns = ostr.str();
162  if (ctxt._fixns.empty())
163  ctxt._fixns = ctxt._systemns;
164  ctxt._clname = prefix;
165 
166  ostr.str("");
167  ostr << type << '.' << major << '.' << minor;
168  ctxt._beginstr = ostr.str();
169 
170  return 0;
171 }

191 void process_value_enums(FieldSpecMap::const_iterator itr, ostream& ost_hpp, ostream& ost_cpp)
192 {
193  string typestr("const ");
194  if (FieldTrait::is_int(itr->second._ftype))
195  typestr += "int ";
196  else if (FieldTrait::is_char(itr->second._ftype))
197  typestr += "char ";
198  else if (FieldTrait::is_float(itr->second._ftype))
199  typestr += "double ";
200  else if (FieldTrait::is_string(itr->second._ftype))
201  typestr += "f8String ";
202  else
203  return;
204 
205  ost_cpp << typestr << itr->second._name << "_realm[] = " << endl << spacer << "{ ";
206  unsigned cnt(0);
207  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
208  {
209  if (cnt)
210  ost_cpp << ", ";
211  ost_cpp << *ditr->first;
212  string transdesc(ditr->second);
213  // replace any illegal c++ identifier characters
214  InPlaceReplaceInSet(ident_set, transdesc, '_');
215  ost_hpp << typestr << itr->second._name << '_';
216  if (transdesc.empty())
217  ost_hpp << *ditr->first;
218  else
219  ost_hpp << transdesc;
220  if (ditr->first->is_range())
221  {
222  if (cnt == 0)
223  ost_hpp << "_lower";
224  else if (cnt == 1)
225  ost_hpp << "_upper";
226  }
227  ost_hpp << '(' << *ditr->first << ");" << endl;
228  ++cnt;
229  }
230  ost_hpp << "const size_t " << itr->second._name << "_realm_els(" << itr->second._dvals->size() << ");" << endl;
231  ost_cpp << " };" << endl;
232 
233  ost_cpp << "const char *" << itr->second._name << "_descriptions[] = " << endl << spacer << "{ ";
234  cnt = 0;
235  for (RealmMap::const_iterator ditr(itr->second._dvals->begin()); ditr != itr->second._dvals->end(); ++ditr)
236  {
237  if (cnt)
238  ost_cpp << ", ";
239  ost_cpp << '"' << ditr->second << '"';
240  ++cnt;
241  }
242  ost_cpp << " };" << endl;
243 }

246 int process_message_fields(const std::string& where, const XmlElement *xt, FieldTraits& fts, const FieldToNumMap& ftonSpec,
247  FieldSpecMap& fspec, const Components& compon)
248 {
249  unsigned processed(0);
250  XmlElement::XmlSet flist;
251  if (xt->find(where, flist))
252  {
253  for(XmlElement::XmlSet::const_iterator fitr(flist.begin()); fitr != flist.end(); ++fitr)
254  {
255  string fname, required;
256  if ((*fitr)->GetAttr("name", fname) && (*fitr)->GetAttr("required", required))
257  {
258  FieldToNumMap::const_iterator ftonItr(ftonSpec.find(fname));
259  FieldSpecMap::iterator fs_itr;
260  if (ftonItr == ftonSpec.end() || (fs_itr = fspec.find(ftonItr->second)) == fspec.end())
261  {
262  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
263  ++glob_errors;
264  continue;
265  }
266 
267  string compname;
268  unsigned compidx((*fitr)->GetAttr("component", compname) ? lookup_component(compon, compname) : 0);
269 
270  // add FieldTrait
271  if (!fts.add(FieldTrait(fs_itr->first, fs_itr->second._ftype, (*fitr)->GetSubIdx(), required == "Y", false, compidx)))
272  {
273  if (!nowarn)
274  cerr << shortName << ':' << recover_line(**fitr) << ": warning: Could not add trait object '" << fname << "' (duplicate ?)" << endl;
275  ++glob_warnings;
276  }
277  else
278  {
279  process_special_traits(fs_itr->first, fts);
280  ++processed;
281  fs_itr->second.set_used();
282  }
283  }
284  else
285  {
286  cerr << shortName << ':' << recover_line(**fitr) << ": error: Field element missing required attributes" << endl;
287  ++glob_errors;
288  }
289  }
290  }
291 
292  return processed;
293 }
