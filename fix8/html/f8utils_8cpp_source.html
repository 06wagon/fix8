80 const string& GetTimeAsStringMS(string& result, Tickval *tv, const unsigned dplaces)
81 {
82  Tickval *startTime, gotTime;
83  if (tv)
84  startTime = tv;
85  else
86  {
87  gotTime.now();
88  startTime = &gotTime;
89  }
90 
91  struct tm tim;
92  time_t tval(startTime->secs());
93  localtime_r(&tval, &tim);
94  ostringstream oss;
95  oss << setfill('0') << setw(4) << (tim.tm_year + 1900) << '-';
96  oss << setw(2) << (tim.tm_mon + 1) << '-' << setw(2) << tim.tm_mday << ' ' << setw(2) << tim.tm_hour;
97  oss << ':' << setw(2) << tim.tm_min << ':';
98  if (dplaces)
99  {
100  const double secs((startTime->secs() % 60) + static_cast<double>(startTime->nsecs()) / Tickval::billion);
101  oss.setf(ios::showpoint);
102  oss.setf(ios::fixed);
103  oss << setw(3 + dplaces) << setfill('0') << setprecision(dplaces) << secs;
104  }
105  else
106  oss << setfill('0') << setw(2) << tim.tm_sec;
107  return result = oss.str();
108 }
