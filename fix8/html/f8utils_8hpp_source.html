738 inline double fast_atof (const char *p)
739 {
740  bool frac(false);
741  double sign(1.), value(0.), scale(1.);
742 
743  while (isspace(*p))
744  ++p;
745 
746  // Get sign, if any.
747  if (*p == '-')
748  {
749  sign = -1.;
750  ++p;
751  }
752  else if (*p == '+')
753  ++p;
754 
755  // Get digits before decimal point or exponent, if any.
756  while (isdigit(*p))
757  {
758  value = value * 10. + (*p - '0');
759  ++p;
760  }
761 
762  // Get digits after decimal point, if any.
763  if (*p == '.')
764  {
765  ++p;
766  double pow10(10.);
767  while (isdigit(*p))
768  {
769  value += (*p - '0') / pow10;
770  pow10 *= 10.;
771  ++p;
772  }
773  }
774 
775  // Handle exponent, if any.
776  if (toupper(*p) == 'E')
777  {
778  unsigned int expon(0);
779  ++p;
780 
781  // Get sign of exponent, if any.
782  if (*p == '-')
783  {
784  frac = true;
785  ++p;
786  }
787  else if (*p == '+')
788  ++p;
789 
790  // Get digits of exponent, if any.
791  while (isdigit(*p))
792  {
793  expon = expon * 10 + (*p - '0');
794  ++p;
795  }
796  if (expon > 308)
797  expon = 308;
798 
799  // Calculate scaling factor.
800  while (expon >= 50)
801  {
802  scale *= 1E50;
803  expon -= 50;
804  }
805  while (expon >= 8)
806  {
807  scale *= 1E8;
808  expon -= 8;
809  }
810  while (expon > 0)
811  {
812  scale *= 10.0;
813  expon -= 1;
814  }
815  }
816 
817  // Return signed and scaled floating point result.
818  return sign * (frac ? (value / scale) : (value * scale));
819 }
