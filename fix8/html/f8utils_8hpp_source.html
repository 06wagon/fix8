544 template<typename T>
545 inline size_t itoa(T value, char *result, const T base=10)
546 {
547  // check that the base if valid
548  if (base < 2 || base > 36)
549  {
550  *result = 0;
551  return 0;
552  }
553 
554  char *ptr(result), *ptr1(result);
555  T tmp_value;
556 
557  do
558  {
559  tmp_value = value;
560  value /= base;
561  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
562  }
563  while (value);
564 
565  // Apply negative sign
566  if (tmp_value < 0)
567  *ptr++ = '-';
568  *ptr-- = 0;
569  while(ptr1 < ptr)
570  {
571  const char tmp_char(*ptr);
572  *ptr-- = *ptr1;
573  *ptr1++ = tmp_char;
574  }
575  return ::strlen(result);
576 }

588 inline double fast_atof (const char *p)
589 {
590  bool frac(false);
591  double sign(1.), value(0.), scale(1.);
592 
593  while (isspace(*p))
594  ++p;
595 
596  // Get sign, if any.
597  if (*p == '-')
598  {
599  sign = -1.;
600  ++p;
601  }
602  else if (*p == '+')
603  ++p;
604 
605  // Get digits before decimal point or exponent, if any.
606  while (isdigit(*p))
607  {
608  value = value * 10. + (*p - '0');
609  ++p;
610  }
611 
612  // Get digits after decimal point, if any.
613  if (*p == '.')
614  {
615  ++p;
616  double pow10(10.);
617  while (isdigit(*p))
618  {
619  value += (*p - '0') / pow10;
620  pow10 *= 10.;
621  ++p;
622  }
623  }
624 
625  // Handle exponent, if any.
626  if (toupper(*p) == 'E')
627  {
628  unsigned int expon(0);
629  ++p;
630 
631  // Get sign of exponent, if any.
632  if (*p == '-')
633  {
634  frac = true;
635  ++p;
636  }
637  else if (*p == '+')
638  ++p;
639 
640  // Get digits of exponent, if any.
641  while (isdigit(*p))
642  {
643  expon = expon * 10 + (*p - '0');
644  ++p;
645  }
646  if (expon > 308)
647  expon = 308;
648 
649  // Calculate scaling factor.
650  while (expon >= 50)
651  {
652  scale *= 1E50;
653  expon -= 50;
654  }
655  while (expon >= 8)
656  {
657  scale *= 1E8;
658  expon -= 8;
659  }
660  while (expon > 0)
661  {
662  scale *= 10.0;
663  expon -= 1;
664  }
665  }
666 
667  // Return signed and scaled floating point result.
668  return sign * (frac ? (value / scale) : (value * scale));
669 }
