658 inline double fast_atof (const char *p)
659 {
660  bool frac(false);
661  double sign(1.), value(0.), scale(1.);
662 
663  while (isspace(*p))
664  ++p;
665 
666  // Get sign, if any.
667  if (*p == '-')
668  {
669  sign = -1.;
670  ++p;
671  }
672  else if (*p == '+')
673  ++p;
674 
675  // Get digits before decimal point or exponent, if any.
676  while (isdigit(*p))
677  {
678  value = value * 10. + (*p - '0');
679  ++p;
680  }
681 
682  // Get digits after decimal point, if any.
683  if (*p == '.')
684  {
685  ++p;
686  double pow10(10.);
687  while (isdigit(*p))
688  {
689  value += (*p - '0') / pow10;
690  pow10 *= 10.;
691  ++p;
692  }
693  }
694 
695  // Handle exponent, if any.
696  if (toupper(*p) == 'E')
697  {
698  unsigned int expon(0);
699  ++p;
700 
701  // Get sign of exponent, if any.
702  if (*p == '-')
703  {
704  frac = true;
705  ++p;
706  }
707  else if (*p == '+')
708  ++p;
709 
710  // Get digits of exponent, if any.
711  while (isdigit(*p))
712  {
713  expon = expon * 10 + (*p - '0');
714  ++p;
715  }
716  if (expon > 308)
717  expon = 308;
718 
719  // Calculate scaling factor.
720  while (expon >= 50)
721  {
722  scale *= 1E50;
723  expon -= 50;
724  }
725  while (expon >= 8)
726  {
727  scale *= 1E8;
728  expon -= 8;
729  }
730  while (expon > 0)
731  {
732  scale *= 10.0;
733  expon -= 1;
734  }
735  }
736 
737  // Return signed and scaled floating point result.
738  return sign * (frac ? (value / scale) : (value * scale));
739 }
