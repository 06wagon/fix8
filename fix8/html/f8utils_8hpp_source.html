533 inline size_t itoa(int value, char *result, const int base=10)
534 {
535  // check that the base if valid
536  if (base < 2 || base > 36)
537  {
538  *result = 0;
539  return 0;
540  }
541 
542  char *ptr(result), *ptr1(result);
543  int tmp_value;
544 
545  do
546  {
547  tmp_value = value;
548  value /= base;
549  *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
550  }
551  while (value);
552 
553  // Apply negative sign
554  if (tmp_value < 0)
555  *ptr++ = '-';
556  *ptr-- = 0;
557  while(ptr1 < ptr)
558  {
559  const char tmp_char(*ptr);
560  *ptr-- = *ptr1;
561  *ptr1++ = tmp_char;
562  }
563  return ::strlen(result);
564 }

576 inline double fast_atof (const char *p)
577 {
578  bool frac(false);
579  double sign(1.), value(0.), scale(1.);
580 
581  while (isspace(*p))
582  ++p;
583 
584  // Get sign, if any.
585  if (*p == '-')
586  {
587  sign = -1.;
588  ++p;
589  }
590  else if (*p == '+')
591  ++p;
592 
593  // Get digits before decimal point or exponent, if any.
594  while (isdigit(*p))
595  {
596  value = value * 10. + (*p - '0');
597  ++p;
598  }
599 
600  // Get digits after decimal point, if any.
601  if (*p == '.')
602  {
603  ++p;
604  double pow10(10.);
605  while (isdigit(*p))
606  {
607  value += (*p - '0') / pow10;
608  pow10 *= 10.;
609  ++p;
610  }
611  }
612 
613  // Handle exponent, if any.
614  if (toupper(*p) == 'E')
615  {
616  unsigned int expon(0);
617  ++p;
618 
619  // Get sign of exponent, if any.
620  if (*p == '-')
621  {
622  frac = true;
623  ++p;
624  }
625  else if (*p == '+')
626  ++p;
627 
628  // Get digits of exponent, if any.
629  while (isdigit(*p))
630  {
631  expon = expon * 10 + (*p - '0');
632  ++p;
633  }
634  if (expon > 308)
635  expon = 308;
636 
637  // Calculate scaling factor.
638  while (expon >= 50)
639  {
640  scale *= 1E50;
641  expon -= 50;
642  }
643  while (expon >= 8)
644  {
645  scale *= 1E8;
646  expon -= 8;
647  }
648  while (expon > 0)
649  {
650  scale *= 10.0;
651  expon -= 1;
652  }
653  }
654 
655  // Return signed and scaled floating point result.
656  return sign * (frac ? (value / scale) : (value * scale));
657 }
