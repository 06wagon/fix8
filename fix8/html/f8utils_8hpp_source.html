715 inline fp_type fast_atof (const char *p)
716 {
717  bool frac(false);
718  fp_type sign(1.), value(0.), scale(1.);
719 
720  while (isspace(*p))
721  ++p;
722 
723  // Get sign, if any.
724  if (*p == '-')
725  {
726  sign = -1.;
727  ++p;
728  }
729  else if (*p == '+')
730  ++p;
731 
732  // Get digits before decimal point or exponent, if any.
733  while (isdigit(*p))
734  {
735  value = value * 10. + (*p - '0');
736  ++p;
737  }
738 
739  // Get digits after decimal point, if any.
740  if (*p == '.')
741  {
742  ++p;
743  fp_type mpow10(10.);
744  while (isdigit(*p))
745  {
746  value += (*p - '0') / mpow10;
747  mpow10 *= 10.;
748  ++p;
749  }
750  }
751 
752  // Handle exponent, if any.
753  if (toupper(*p) == 'E')
754  {
755  unsigned int expon(0);
756  ++p;
757 
758  // Get sign of exponent, if any.
759  if (*p == '-')
760  {
761  frac = true;
762  ++p;
763  }
764  else if (*p == '+')
765  ++p;
766 
767  // Get digits of exponent, if any.
768  while (isdigit(*p))
769  {
770  expon = expon * 10 + (*p - '0');
771  ++p;
772  }
773 #if defined USE_SINGLE_PRECISION
774  if (expon > 38)
775  expon = 38;
776 #else
777  if (expon > 308)
778  expon = 308;
779 
780  // Calculate scaling factor.
781  while (expon >= 50)
782  {
783  scale *= 1E50;
784  expon -= 50;
785  }
786 #endif
787 
788  while (expon >= 8)
789  {
790  scale *= 1E8;
791  expon -= 8;
792  }
793  while (expon > 0)
794  {
795  scale *= 10.0;
796  expon -= 1;
797  }
798  }
799 
800  // Return signed and scaled floating point result.
801  return sign * (frac ? (value / scale) : (value * scale));
802 }
