45  /* FF_FARMA creates an accelerator farm with 'nworkers' workers.
46  * The functions called in each worker must have the following
47  * signature:
48  * bool (*F)(type &)
49  * If the function F returns false an EOS is produced in the
50  * output channel.
51  */
52 
53 #define FF_FARMA(farmname, F, type, nworkers, qlen) \
54  class _FF_node_##farmname_##F: public ff_node { \
55  public: \
56  void* svc(void* task) { \
57  type* _ff_in = (type*)task; \
58  if (!F(*_ff_in)) return NULL; \
59  return task; \
60  } \
61  }; \
62  ff_farm<> _FF_node_##farmname(true,qlen,qlen,true,nworkers,true); \
63  _FF_node_##farmname.cleanup_workers(); \
64  _FF_node_##farmname.set_scheduling_ondemand(); \
65  std::vector<ff_node*> w_##farmname; \
66  for(int i=0;i<nworkers;++i) \
67  w_##farmname.push_back(new _FF_node_##farmname_##F); \
68  _FF_node_##farmname.add_workers(w_##farmname); \
69  _FF_node_##farmname.add_collector(NULL);
