67 #if defined SLEEP_NO_YIELD

132 // C++ wrapper around fastflow atomic
133 template<typename T>
134 class ff_atomic
135 {
136  mutable atomic_long_t _rep;
137 
138 public:
139  typedef T value_type;
140 
141  value_type operator=(const value_type rhs)
142  { atomic_long_set(&_rep, rhs); return rhs; }
143 
144  ff_atomic<value_type>& operator=(const ff_atomic<value_type>& rhs)
145  { atomic_long_set(&_rep, atomic_long_read(&rhs._rep)); return *this; }
146 
147  value_type operator++(int) // postfix
148  { value_type v(atomic_long_read(&_rep)); atomic_long_inc(&_rep); return v; }
149  value_type operator--(int) // postfix
150  { value_type v(atomic_long_read(&_rep)); atomic_long_dec(&_rep); return v; }
151  value_type operator++() // prefix
152  { return atomic_long_inc_return(&_rep); }
153  value_type operator--() // prefix
154  { return atomic_long_dec_return(&_rep); }
155  value_type operator+=(value_type value)
156  { atomic_long_add(value, &_rep); return atomic_long_read(&_rep); }
157  value_type operator-=(value_type value)
158  { atomic_long_sub(value, &_rep); return atomic_long_read(&_rep); }
159 
160  operator value_type() { return static_cast<T>(atomic_long_read(&_rep)); }
161  operator value_type() const { return static_cast<T>(atomic_long_read(&_rep)); }
162 };
