857 {

860  31 + 28,

863  31 + 28 + 31 + 30 + 31,

868  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31,

870  31 + 28 + 31 + 30 + 31 + 30 + 31 + 31 + 30 + 31 + 30 + 31

878 template<const unsigned short field>
879 inline Tickval::ticks Field<UTCTimestamp, field>::datetimeparse(const char *ptr, size_t len)
880 {
881  Tickval::ticks result(Tickval::noticks);
882  int millisecond(0);
883  tm tms = {};
884 
885  ptr += parseDate(ptr, 4, tms.tm_year);
886  tms.tm_year -= 1900;
887  ptr += parseDate(ptr, 2, tms.tm_mon);
888  --tms.tm_mon;
889  ptr += parseDate(ptr, 2, tms.tm_mday);
890  ptr += parseDate(++ptr, 2, tms.tm_hour);
891  ptr += parseDate(++ptr, 2, tms.tm_min);
892  ptr += parseDate(++ptr, 2, tms.tm_sec);
893  switch(len)
894  {
895  case _with_ms: // 19981231-23:59:59.123
896  ptr += parseDate(++ptr, 3, millisecond);
897  result = millisecond * Tickval::million; // drop through
898  case _sec_only: // 19981231-23:59:59
899  result += time_to_epoch(tms) * Tickval::billion;
900  break;
901  default:
902  break;
903  }
904 
905  return result;
906 }
