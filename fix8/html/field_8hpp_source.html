977 {
978  if (len == 0 || (*ptr == 'n' && len == 3 && *(ptr + 1) == 'o' && *(ptr + 2) == 'w')) // special cases initialise to 'now'
979  return Tickval(true).get_ticks();
980 
981  Tickval::ticks result(Tickval::noticks);
982  int millisecond(0);
983  tm tms {};
984 
985  ptr += parse_decimal(ptr, 2, tms.tm_hour);
986  ++ptr;
987  ptr += parse_decimal(ptr, 2, tms.tm_min);
988  ++ptr;
989  ptr += parse_decimal(ptr, 2, tms.tm_sec);
990  switch(len)
991  {
992  case 12: // 23:59:59.123
993  parse_decimal(++ptr, 3, millisecond);
994  result = millisecond * Tickval::million; // drop through
995  case 8: // 23:59:59
996  if (!timeonly)
997  result += time_to_epoch(tms) * Tickval::billion;
998  else
999  result += (tms.tm_hour * 3600ULL + tms.tm_min * 60ULL + tms.tm_sec) * Tickval::billion;
1000  break;
1001  default:
1002  break;
1003  }
1004 
1005  return result;
