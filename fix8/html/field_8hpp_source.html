677 {
678  static const int mon_days[] = {0,

681  31 + 28 + 31,

694  const int tdays(mon_days[ltm.tm_mon] + (ltm.tm_mday ? ltm.tm_mday - 1 : 0) + tyears * 365 + (tyears + 2) / 4);

711 inline size_t date_time_format(const Tickval& tickval, char *to, const TimeIndicator ind)
712 {
713  tm result;
714  tickval.as_tm(result);
715  const char *start(to);
716 
717  if (ind > _time_with_ms)
718  {
719  format0(result.tm_year + 1900, to, 4);
720  to += 4;
721  format0(result.tm_mon + 1, to, 2);
722  to += 2;
723  if (ind == _short_date_only)
724  return to - start;
725  format0(result.tm_mday, to, 2);
726  to += 2;
727  if (ind == _date_only)
728  return to - start;
729  *to++ = '-';
730  }
731 
732  format0(result.tm_hour, to, 2);
733  to += 2;
734  *to++ = ':';
735  format0(result.tm_min, to, 2);
736  to += 2;
737  *to++ = ':';
738  format0(result.tm_sec, to, 2);
739  to += 2;
740 
741  if (ind == _time_with_ms || ind == _with_ms)
742  {
743  *to++ = '.';
744  format0(tickval.msecs(), to, 3);
745  to += 3;
746  }
747 
748  return to - start;
749 }

755 inline Tickval::ticks date_time_parse(const char *ptr, size_t len)
756 {
757  Tickval::ticks result(Tickval::noticks);
758  int millisecond(0);
759  tm tms = {};
760 
761  ptr += parse_decimal(ptr, 4, tms.tm_year);
762  tms.tm_year -= 1900;
763  ptr += parse_decimal(ptr, 2, tms.tm_mon);
764  --tms.tm_mon;
765  ptr += parse_decimal(ptr, 2, tms.tm_mday);
766  ++ptr;
767  ptr += parse_decimal(ptr, 2, tms.tm_hour);
768  ++ptr;
769  ptr += parse_decimal(ptr, 2, tms.tm_min);
770  ++ptr;
771  ptr += parse_decimal(ptr, 2, tms.tm_sec);
772  switch(len)
773  {
774  case 21: //_with_ms: // 19981231-23:59:59.123
775  parse_decimal(++ptr, 3, millisecond);
776  result = millisecond * Tickval::million; // drop through

780  default:

782  }
