71  template<typename T>
72  bool is_valid(const T& what) const
73  {
74  const T *rng(static_cast<const T*>(_range));
75  return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);
76  }
77 
82  template<typename T>
83  const T& get_rlm_val(const int idx) const { return *(static_cast<const T*>(_range) + idx); }
84 
89  template<typename T>
90  int get_rlm_idx(const T& what) const
91  {
92  if (_dtype == dt_set)
93  {
94  const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));
95  return res != rng + _sz ? res - rng : -1;
96  }
97  return 0;
98  }

794 template<const unsigned short field>
795 inline size_t Field<UTCTimestamp, field>::DateTimeFormat(const Poco::DateTime& dateTime, char *to, const MillisecondIndicator ind) const
796 {
797  format0(dateTime.year(), to, 4);
798  format0(dateTime.month(), to + 4, 2);
799  format0(dateTime.day(), to + 6, 2);
800  to[8] = '-';
801 
802  format0(dateTime.hour(), to + 9, 2);
803  to[11] = ':';
804 
805  format0(dateTime.minute(), to + 12, 2);
806  to[14] = ':';
807 
808  format0(dateTime.second(), to + 15, 2);
809 
810  if(ind != _with_ms)
811  return _sec_only; //length of "YYYYMMDD-HH:MM:SS"
812 
813  to[17] = '.';
814  format0(dateTime.millisecond(), to + 18, 3);
815  return _with_ms; //length of "YYYYMMDD-HH:MM:SS.MMM"
816 }

818 template<const unsigned short field>
819 inline void Field<UTCTimestamp, field>::DateTimeParse(const std::string& from, Poco::DateTime& dateTime, const MillisecondIndicator ind) const
820 {
821  std::string::const_iterator it(from.begin());
822  short year(0), month(0), day(0), hour(0), minute(0), second(0), millisecond(0);
823 
824  parseDate(it, 4, year);
825  parseDate(it, 2, month);
826  parseDate(it, 2, day);
827  parseDate(++it, 2, hour);
828  parseDate(++it, 2, minute);
829  parseDate(++it, 2, second);
830 
831  if(ind == _with_ms)
832  parseDate(++it, 3, millisecond);
833 
834  if (Poco::DateTime::isValid(year, month, day, hour, minute, second, millisecond))
835  dateTime.assign(year, month, day, hour, minute, second, millisecond);
836 }
