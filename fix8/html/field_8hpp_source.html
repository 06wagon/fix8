57 struct RealmBase
58 {
59  enum RealmType { dt_range, dt_set };
60 
61  const void *_range;
62  RealmType _dtype;
63  FieldTrait::FieldType _ftype;
64  const int _sz;
65  const char * const *_descriptions;
66 
71  template<typename T>
72  bool is_valid(const T& what) const
73  {
74  const T *rng(static_cast<const T*>(_range));
75  return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);
76  }
77 
82  template<typename T>
83  const T& get_rlm_val(const int idx) const
84  {
85  return *(static_cast<const T*>(_range) + idx);
86  }
87 
92  template<typename T>
93  int get_rlm_idx(const T& what) const
94  {
95  if (_dtype == dt_set)
96  {
97  const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));
98  return res != rng + _sz ? res - rng : -1;
99  }
100  return 0;
101  }
