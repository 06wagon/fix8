967 inline Tickval::ticks time_parse(const char *ptr, size_t len, bool timeonly=false)
968 {
969  if (len == 0 || *ptr == '!') // special case initialise to 'now'
970  return Tickval(true).get_ticks();
971 
972  Tickval::ticks result(Tickval::noticks);
973  int millisecond(0);
974  tm tms {};
975 
976  ptr += parse_decimal(ptr, 2, tms.tm_hour);
977  ++ptr;
978  ptr += parse_decimal(ptr, 2, tms.tm_min);
979  ++ptr;
980  ptr += parse_decimal(ptr, 2, tms.tm_sec);
981  switch(len)
982  {
983  case 12: // 23:59:59.123
984  parse_decimal(++ptr, 3, millisecond);
985  result = millisecond * Tickval::million; // drop through
986  case 8: // 23:59:59
987  if (!timeonly)
988  result += time_to_epoch(tms) * Tickval::billion;
989  else
990  result += (tms.tm_hour * 3600ULL + tms.tm_min * 60ULL + tms.tm_sec) * Tickval::billion;
991  break;
992  default:
993  break;
994  }
995 
996  return result;
997 }
