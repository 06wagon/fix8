761 inline size_t date_time_format(const Tickval& tickval, char *to, const TimeIndicator ind)
762 {

764  tickval.as_tm(result);
765  const char *start(to);
766 
767  if (ind > _time_with_ms)
768  {
769  format0(result.tm_year + 1900, to, 4);

771  format0(result.tm_mon + 1, to, 2);
772  to += 2;
773  if (ind == _short_date_only)

775  format0(result.tm_mday, to, 2);
776  to += 2;
777  if (ind == _date_only)
778  return to - start;

784  *to++ = ':';
785  format0(result.tm_min, to, 2);

788  format0(result.tm_sec, to, 2);

805 inline Tickval::ticks date_time_parse(const char *ptr, size_t len)
806 {
807  if (*ptr == '!') // special case initialise to 'now'
808  return Tickval(true).get_ticks();
809 
810  Tickval::ticks result(Tickval::noticks);
811  int millisecond(0);
812  tm tms {};
813 
814  ptr += parse_decimal(ptr, 4, tms.tm_year);
815  tms.tm_year -= 1900;
816  ptr += parse_decimal(ptr, 2, tms.tm_mon);
817  --tms.tm_mon;
818  ptr += parse_decimal(ptr, 2, tms.tm_mday);
819  ++ptr;
820  ptr += parse_decimal(ptr, 2, tms.tm_hour);
821  ++ptr;
822  ptr += parse_decimal(ptr, 2, tms.tm_min);
823  ++ptr;
824  ptr += parse_decimal(ptr, 2, tms.tm_sec);
825  switch(len)
826  {
827  case 21: //_with_ms: // 19981231-23:59:59.123
828  parse_decimal(++ptr, 3, millisecond);
829  result = millisecond * Tickval::million; // drop through
830  case 17: //: // 19981231-23:59:59
831  result += time_to_epoch(tms) * Tickval::billion;
832  break;
833  default:
834  break;
835  }
