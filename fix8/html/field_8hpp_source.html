74  template<typename T>
75  bool is_valid(const T& what) const
76  {
77  const T *rng(static_cast<const T*>(_range));
78  return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);
79  }
80 
85  template<typename T>
86  const T& get_rlm_val(const int idx) const
87  {
88  return *(static_cast<const T*>(_range) + idx);
89  }
90 
95  template<typename T>
96  int get_rlm_idx(const T& what) const
97  {
98  if (_dtype == dt_set)
99  {
100  const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));
101  return res != rng + _sz ? res - rng : -1;
102  }
103  return 0;
104  }
