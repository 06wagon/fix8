73  template<typename T>
74  bool is_valid(const T& what) const
75  {
76  const T *rng(static_cast<const T*>(_range));
77  return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);
78  }
79 
84  template<typename T>
85  const T& get_rlm_val(const int idx) const
86  {
87  return *(static_cast<const T*>(_range) + idx);

94  template<typename T>
95  int get_rlm_idx(const T& what) const
96  {
97  if (_dtype == dt_set)
98  {
99  const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));
100  return res != rng + _sz ? res - rng : -1;
101  }
102  return 0;
103  }

802 template<const unsigned short field>
803 inline size_t Field<UTCTimestamp, field>::DateTimeFormat(const Poco::DateTime& dateTime, char *to, const MillisecondIndicator ind) const
804 {
805  format0(dateTime.year(), to, 4);
806  format0(dateTime.month(), to + 4, 2);
807  format0(dateTime.day(), to + 6, 2);
808  to[8] = '-';
809 
810  format0(dateTime.hour(), to + 9, 2);
811  to[11] = ':';
812 
813  format0(dateTime.minute(), to + 12, 2);
814  to[14] = ':';
815 
816  format0(dateTime.second(), to + 15, 2);
817 
818  if(ind != _with_ms)
819  return _sec_only; //length of "YYYYMMDD-HH:MM:SS"
820 
821  to[17] = '.';
822  format0(dateTime.millisecond(), to + 18, 3);
823  return _with_ms; //length of "YYYYMMDD-HH:MM:SS.MMM"
824 }

826 template<const unsigned short field>
827 inline void Field<UTCTimestamp, field>::DateTimeParse(const std::string& from, Poco::DateTime& dateTime, const MillisecondIndicator ind) const
828 {
829  std::string::const_iterator it(from.begin());
830  short year(0), month(0), day(0), hour(0), minute(0), second(0), millisecond(0);
831 
832  parseDate(it, 4, year);
833  parseDate(it, 2, month);
834  parseDate(it, 2, day);
835  parseDate(++it, 2, hour);
836  parseDate(++it, 2, minute);
837  parseDate(++it, 2, second);
838 
839  if(ind == _with_ms)
840  parseDate(++it, 3, millisecond);
841 
842  if (Poco::DateTime::isValid(year, month, day, hour, minute, second, millisecond))
843  dateTime.assign(year, month, day, hour, minute, second, millisecond);
844 }
