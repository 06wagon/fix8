75  template<typename T>

77  {
78  const T *rng(static_cast<const T*>(_range));
79  return _dtype == dt_set ? std::binary_search(rng, rng + _sz, what) : *rng <= what && what <= *(rng + 1);

96  template<typename T>
97  int get_rlm_idx(const T& what) const
98  {
99  if (_dtype == dt_set)
100  {
101  const T *rng(static_cast<const T*>(_range)), *res(std::lower_bound(rng, rng + _sz, what));

103  }
104  return 0;
105  }
