66 bool FilePersister::initialise(const f8String& dbDir, const f8String& dbFname)
67 {
68  if (_opened)
69  return true;
70 
71  f8String odbdir(dbDir);
72  ostringstream ostr;
73  ostr << CheckAddTrailingSlash(odbdir) << dbFname;
74  _dbFname = ostr.str();
75  ostr << ".idx";
76  _dbIname = ostr.str();
77 
78  if (!exist(_dbFname))
79  {
80  if ((_fod = open(_dbFname.c_str(), O_RDWR | O_CREAT, 0600)) < 0)
81  {
82  ostringstream eostr;
83  eostr << "Error creating database: " << _dbFname << " (" << strerror(errno) << ')';
84  GlobalLogger::log(eostr.str());
85  return false;
86  }
87  if ((_iod = open(_dbIname.c_str(), O_RDWR | O_CREAT | O_TRUNC, 0600)) < 0)
88  {
89  ostringstream eostr;
90  eostr << "Error creating database index: " << _dbIname << " (" << strerror(errno) << ')';
91  GlobalLogger::log(eostr.str());
92  return false;
93  }
94 
95  _wasCreated = true;
96  }
97  else
98  {
99  if ((_fod = open(_dbFname.c_str(), O_RDWR)) < 0)
100  {
101  ostringstream eostr;
102  eostr << "Error opening existing database: " << _dbFname << " (" << strerror(errno) << ')';
103  GlobalLogger::log(eostr.str());
104  return false;
105  }
106  if ((_iod = open(_dbIname.c_str(), O_RDWR)) < 0)
107  {
108  ostringstream eostr;
109  eostr << "Error opening existing database index: " << _dbIname << " (" << strerror(errno) << ')';
110  GlobalLogger::log(eostr.str());
111  return false;
112  }
113 
114  IPrec iprec;
115  while (true)
116  {
117  ssize_t blrd(read(_iod, static_cast<void *>(&iprec), sizeof(IPrec)));
118  if (blrd < 0)
119  {
120  ostringstream eostr;
121  eostr << "Error reading existing database index: " << _dbIname << " (" << strerror(errno) << ')';
122  GlobalLogger::log(eostr.str());
123  return false;
124  }
125  else if (blrd == 0)
126  break; // eof
127 
128  if (!_index.insert(Index::value_type(iprec._seq, iprec._prec)).second)
129  {
130  ostringstream eostr;
131  eostr << "Error inserting index record into database index: " << _dbIname << " (idx=" << iprec._seq << ')';
132  GlobalLogger::log(eostr.str());
133  return false;
134  }
135  }
136 
137  unsigned last;
138  if (get_last_seqnum(last))
139  {
140  ostringstream ostr;
141  ostr << _dbFname << ": Last sequence is " << last;
142  GlobalLogger::log(ostr.str());
143  }
144  }
145 
146  return _opened = true;
147 }

163 unsigned FilePersister::get(const unsigned from, const unsigned to, Session& session,
164  bool (Session::*callback)(const Session::SequencePair& with, Session::RetransmissionContext& rctx)) const
165 {
166  unsigned last_seq(0);
167  get_last_seqnum(last_seq);
168  unsigned recs_sent(0), startSeqNum(find_nearest_highest_seqnum (from, last_seq));
169  const unsigned finish(to == 0 ? last_seq : to);
170  Session::RetransmissionContext rctx(from, to, session.get_next_send_seq());
171 
172  if (!startSeqNum || from > finish)
173  {
174  GlobalLogger::log("No records found");
175  rctx._no_more_records = true;
176  (session.*callback)(Session::SequencePair(0, ""), rctx);
177  return 0;
178  }
179 
180  Index::const_iterator itr(_index.find(startSeqNum));
181  if (itr != _index.end())
182  {
183  char buff[MAX_MSG_LENGTH];
184 
185  do
186  {
187  if (!itr->first || itr->first > finish)
188  break;
189  if (lseek(_fod, itr->second._offset, SEEK_SET) < 0)
190  {
191  ostringstream eostr;
192  eostr << "Error could not seek to correct index location for get: " << _dbFname;
193  GlobalLogger::log(eostr.str());
194  break;
195  }
196 
197  if (read (_fod, buff, itr->second._size) != itr->second._size)
198  {
199  ostringstream eostr;
200  eostr << "Error could not read message record for seqnum " << itr->first << " from: " << _dbFname;
201  GlobalLogger::log(eostr.str());
202  break;
203  }
204 
205  Session::SequencePair txresult(itr->first, f8String(buff, itr->second._size));
206  ++recs_sent;
207  if (!(session.*callback)(txresult, rctx))
208  break;
209  }
210  while(++itr != _index.end());
211 
212  rctx._no_more_records = true;
213  (session.*callback)(Session::SequencePair(0, ""), rctx);
214  }
215  else
216  {
217  ostringstream ostr;
218  ostr << "record not found (" << startSeqNum << ')';
219  GlobalLogger::log(ostr.str());
220  }
221 
222  return recs_sent;
223 }

226 bool FilePersister::put(const unsigned sender_seqnum, const unsigned target_seqnum)
227 {
228  if (!_opened)
229  return false;
230  IPrec iirec(0, sender_seqnum, target_seqnum);
231  if (_index.empty())
232  _index.insert(Index::value_type(0, iirec._prec));
233  else
234  _index[0] = iirec._prec;
235 
236  if (lseek(_iod, 0, SEEK_SET) < 0)
237  {
238  ostringstream eostr;
239  eostr << "Error could not seek to 0 for seqnum persitence: " << _dbIname;
240  GlobalLogger::log(eostr.str());
241  return false;
242  }
243  return write (_iod, static_cast<void *>(&iirec), sizeof(iirec)) == sizeof(iirec);
244 }

247 bool FilePersister::put(const unsigned seqnum, const f8String& what)
248 {
249  if (!_opened || !seqnum)
250  return false;
251 
252  if (_index.find(seqnum) != _index.end())
253  {
254  ostringstream eostr;
255  eostr << "Error seqnum " << seqnum << " already persisted in: " << _dbIname;
256  GlobalLogger::log(eostr.str());
257  return false;
258  }
259  if (lseek(_iod, 0, SEEK_END) < 0)
260  {
261  ostringstream eostr;
262  eostr << "Error could not seek to index end for seqnum persitence: " << _dbIname;
263  GlobalLogger::log(eostr.str());
264  return false;
265  }
266  off_t offset;
267  if ((offset = lseek(_fod, 0, SEEK_END)) < 0)
268  {
269  ostringstream eostr;
270  eostr << "Error could not seek to end for seqnum persitence: " << _dbFname;
271  GlobalLogger::log(eostr.str());
272  return false;
273  }
274  IPrec iprec(seqnum, offset, what.size());
275  if (write (_iod, static_cast<void *>(&iprec), sizeof(IPrec)) != sizeof(IPrec))
276  {
277  ostringstream eostr;
278  eostr << "Error could not write index record for seqnum " << seqnum << " to: " << _dbIname;
279  GlobalLogger::log(eostr.str());
280  return false;
281  }
282  if (write (_fod, what.data(), sizeof(what.size())) != sizeof(what.size()))
283  {
284  ostringstream eostr;
285  eostr << "Error could not write record for seqnum " << seqnum << " to: " << _dbFname;
286  GlobalLogger::log(eostr.str());
287  return false;
288  }
289 
290  return _index.insert(Index::value_type(seqnum, iprec._prec)).second;
291 }

357 unsigned FilePersister::find_nearest_highest_seqnum (const unsigned requested, const unsigned last) const
358 {
359  if (last)
360  {
361  for (unsigned startseqnum(requested); startseqnum <= last; ++startseqnum)
362  {
363  Index::const_iterator itr(_index.find(startseqnum));

367  }
