356 bool tex_router_server::operator() (const TEX::NewOrderSingle *msg) const
357 {
358  static unsigned oid(0), eoid(0);
359 
360  TEX::OrderQty qty;
361  TEX::Price price;
362 
363  msg->get(qty);
364  msg->get(price);
365 
366  TEX::ExecutionReport *er(new TEX::ExecutionReport);
367  msg->copy_legal(er);
368  if (!quiet)
369  cout << endl;
370 
371  ostringstream oistr;
372  oistr << "ord" << ++oid;
373  *er += new TEX::OrderID(oistr.str());
374  *er += new TEX::ExecType(TEX::ExecType_NEW);
375  *er += new TEX::OrdStatus(TEX::OrdStatus_NEW);
376  *er += new TEX::LeavesQty(qty());
377  *er += new TEX::CumQty(0);
378  *er += new TEX::AvgPx(0);
379  *er += new TEX::LastCapacity('1');
380  *er += new TEX::ReportToExch('Y');
381  *er += new TEX::ExecID(oistr.str());
382  _session.send(er);
383 
384  unsigned remaining_qty(qty()), cum_qty(0);
385  while (remaining_qty > 0)
386  {
387  unsigned trdqty(RandDev::getrandom(remaining_qty));
388  if (!trdqty)
389  trdqty = 1;
390  er = new TEX::ExecutionReport;
391  msg->copy_legal(er);
392  *er += new TEX::OrderID(oistr.str());
393  ostringstream eistr;
394  eistr << "exec" << ++eoid;
395  *er += new TEX::ExecID(eistr.str());
396  *er += new TEX::ExecType(TEX::ExecType_NEW);
397  *er += new TEX::OrdStatus(remaining_qty == trdqty ? TEX::OrdStatus_FILLED : TEX::OrdStatus_PARTIALLY_FILLED);
398  remaining_qty -= trdqty;
399  cum_qty += trdqty;
400  *er += new TEX::LeavesQty(remaining_qty);
401  *er += new TEX::CumQty(cum_qty);
402  *er += new TEX::LastQty(trdqty);
403  *er += new TEX::AvgPx(price());
404  _session.send(er);
405  }
406 
407  return true;
408 }
