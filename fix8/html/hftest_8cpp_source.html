465 {
466  unsigned snt(0);

483  coro_yield;
484  }
485  }
486  ses->get_connection()->set_tcp_cork_flag(false);
487  return _session.cached();
488 }

585 bool tex_router_server::operator() (const FIX8::TEX::NewOrderSingle *msg)
586 {
587  static unsigned oid(0), eoid(0);
588  FIX8::TEX::OrderQty qty;
589  FIX8::TEX::Price price;
590  msg->get(qty);
591  msg->get(price);
592 
593  FIX8::TEX::ExecutionReport *er(new FIX8::TEX::ExecutionReport);
594  msg->copy_legal(er);
595 
596  ostringstream oistr;
597  oistr << "ord" << ++oid;
598  *er << new FIX8::TEX::OrderID(oistr.str())
599  << new FIX8::TEX::ExecType(FIX8::TEX::ExecType_NEW);
600  unsigned ordResult(RandDev::getrandom(3));
601  switch (ordResult)
602  {
603  default:
604  case 0:
605  *er << new FIX8::TEX::OrdStatus(FIX8::TEX::OrdStatus_NEW);
606  break;
607  case 1:
608  *er << new FIX8::TEX::OrdStatus(FIX8::TEX::OrdStatus_CANCELED);
609  break;
610  case 2:
611  *er << new FIX8::TEX::OrdStatus(FIX8::TEX::OrdStatus_REJECTED);
612  break;
613  }
614 
615  *er << new FIX8::TEX::LeavesQty(qty())
616  << new FIX8::TEX::CumQty(0.)
617  << new FIX8::TEX::AvgPx(0.)
618  << new FIX8::TEX::LastCapacity('5')
619  << new FIX8::TEX::ReportToExch('Y')
620  << new FIX8::TEX::ExecTransType(FIX8::TEX::ExecTransType_NEW)
621  << new FIX8::TEX::ExecID(oistr.str());
622 
623  _session.send(er);
624 
625  if (ordResult == 0)
626  {
627  unsigned remaining_qty(qty()), cum_qty(0);
628  while (remaining_qty > 0)
629  {
630  unsigned trdqty(1 + RandDev::getrandom(remaining_qty));
631  remaining_qty -= trdqty;
632  cum_qty += trdqty;
633  FIX8::TEX::ExecutionReport *ner(new FIX8::TEX::ExecutionReport);
634  msg->copy_legal(ner);
635  ostringstream eistr;
636  eistr << "exec" << ++eoid;
637 
638  *ner << new FIX8::TEX::ExecID(eistr.str())
639  << new FIX8::TEX::OrderID(oistr.str())
640  << new FIX8::TEX::ExecType(FIX8::TEX::ExecType_NEW)
641  << new FIX8::TEX::OrdStatus(remaining_qty == trdqty ? FIX8::TEX::OrdStatus_FILLED : FIX8::TEX::OrdStatus_PARTIALLY_FILLED)
642  << new FIX8::TEX::LeavesQty(remaining_qty)
643  << new FIX8::TEX::CumQty(cum_qty)
644  << new FIX8::TEX::ExecTransType(FIX8::TEX::ExecTransType_NEW)
645  << new FIX8::TEX::AvgPx(price());
646 
647  _session.send(ner);
648  }
649  }
650 
651  return true;
652 }
