533  virtual void * svc(void *) {
534  void * task = NULL;
535  void * ret = (void *)FF_EOS;
536  bool inpresent = (get_in_buffer() != NULL);
537  bool skipfirstpop = skip1pop;
538 
539  // the following case is possible when the emitter is a dnode
540  if (!inpresent && filter && (filter->get_in_buffer()!=NULL)) {
541  inpresent = true;
542  set_in_buffer(filter->get_in_buffer());
543  }
544 
545  gettimeofday(&wtstart,NULL);
546  if (!master_worker && (multi_input_size<=0)) {
547  do {
548  if (inpresent) {
549  if (!skipfirstpop) pop(&task);
550  else skipfirstpop=false;
551 
552  if (task == (void*)FF_EOS) {
553  if (filter) filter->eosnotify();
554  push_eos();
555  break;
556  } else if (task == (void*)FF_EOS_NOFREEZE) {
557  if (filter) filter->eosnotify();
558  push_eos(true);
559  ret = task;
560  break;
561  }
562  }
563 
564  if (filter) {
565  FFTRACE(register ticks t0 = getticks());
566 
567  task = filter->svc(task);
568 
569 #if defined(TRACE_FASTFLOW)
570  register ticks diff=(getticks()-t0);
571  tickstot +=diff;
572  ticksmin=(std::min)(ticksmin,diff);
573  ticksmax=(std::max)(ticksmax,diff);
574 #endif
575  if (task == GO_ON) continue;
576 
577  // if the filter returns NULL we exit immediatly
578  if (task ==(void*)FF_EOS_NOFREEZE) {
579  push_eos(true);
580  ret = task;
581  break;
582  }
583  if (!task || task==(void*)FF_EOS) {
584  push_eos();
585  ret = (void*)FF_EOS;
586  break;
587  }
588  } else
589  if (!inpresent) { push_goon(); push_eos(); ret=(void*)FF_EOS; break;}
590 
591  schedule_task(task);
592  } while(true);
593  } else {
594  int nw=0, neos=0;
595  // contains current worker
596  std::deque<ff_node *> availworkers;
597 
598  //assert( master_worker ^ ( multi_input != NULL) );
599 
600  if (master_worker) {
601  for(int i=0;i<running;++i)
602  availworkers.push_back(workers[i]);
603  nw = running;
604  }
605  if (multi_input) {
606  multi_input_start = availworkers.size();
607  for(int i=0;i<multi_input_size;++i)
608  availworkers.push_back(multi_input[i]);
609  nw += multi_input_size;
610  }
611  std::deque<ff_node *>::iterator start(availworkers.begin());
612  std::deque<ff_node *>::iterator victim(availworkers.begin());
613  do {
614  if (!skipfirstpop) {
615  victim=collect_task(&task, availworkers, start);
616  } else skipfirstpop=false;
617 
618  if ((task == (void*)FF_EOS) ||
619  (task == (void*)FF_EOS_NOFREEZE)) {
620  if (channelid == -1) ++neos;
621  if ((victim == availworkers.end()) ||
622  (neos == multi_input_size) )
623  push_eos((task==(void*)FF_EOS_NOFREEZE));
624  else {
625  if (filter) filter->eosnotify(channelid);
626  availworkers.erase(victim);
627  start=availworkers.begin(); // restart iterator
628  }
629 
630  if (!--nw) {
631  ret = task;
632  break; // received all EOS, exit
633  }
634  } else {
635  if (filter) {
636  FFTRACE(register ticks t0 = getticks());
637 
638  task = filter->svc(task);
639 
640 #if defined(TRACE_FASTFLOW)
641  register ticks diff=(getticks()-t0);
642  tickstot +=diff;
643  ticksmin=(std::min)(ticksmin,diff);
644  ticksmax=(std::max)(ticksmax,diff);
645 #endif
646 
647  if (task == GO_ON) continue;
648 
649  // if the filter returns NULL we exit immediatly
650  if (task ==(void*)FF_EOS_NOFREEZE) {
651  push_eos(true);
652  ret = task;
653  break;
654  }
655  if (!task || (task==(void*)FF_EOS)) {
656  push_eos();
657  ret = (void*)FF_EOS;
658  break;
659  }
660  }
661  schedule_task(task);
662  }
663  } while(1);
664  }
665  gettimeofday(&wtstop,NULL);
666  wttime+=diffmsec(wtstop,wtstart);
667 
668  return ret;
669  }
