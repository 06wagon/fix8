75 int Logger::operator()()
76 {
77  unsigned received(0);
78  bool stopping(false);
79 
80  for (;;)
81  {
82  LogElement msg;
83  if (stopping) // make sure we dequeue any pending msgs before exiting
84  {
85  if (!_msg_queue.try_pop(msg))
86  break;
87  }
88  else
89  _msg_queue.pop (msg); // will block
90 
91  ++received;
92 
93  if (msg._str.empty()) // means exit
94  {
95  stopping = true;
96  continue;
97  }
98 
99  ostringstream ostr;
100 
101  if (_flags & sequence)
102  {
103  ostr << setw(7) << right << setfill('0');
104  if (_flags & direction)
105  ostr << (msg._val ? ++_sequence : ++_osequence) << ' ';
106  else
107  ostr << ++_sequence << ' ';
108  }
109 
110  if (_flags & thread)
111  ostr << get_thread_code(msg._tid) << ' ';
112 
113  if (_flags & direction)
114  ostr << (msg._val ? " in" : "out") << ' ' ;
115 
116  if (_flags & timestamp)
117  {
118  string ts;
119  ostr << GetTimeAsStringMS(ts, &msg._when, 9) << ' ';
120  }
121 
122  if (_flags & buffer)
123  {
124  string result(ostr.str());
125  result += msg._str;
126  _buffer.push_back(result);
127  }
128  else
129  {
130  tbb::mutex::scoped_lock guard(_mutex);
131  get_stream() << ostr.str() << msg._str << endl;
132  }
133  }
134 
135  return 0;
136 }

149 void Logger::purge_thread_codes()
150 {
151  tbb::mutex::scoped_lock guard(_mutex);
152 
153  for (ThreadCodes::iterator itr(_thread_codes.begin()); itr != _thread_codes.end();)
154  {
155  // a little trick to see if a thread is still alive
156  clockid_t clock_id;
157  if (pthread_getcpuclockid(itr->first, &clock_id) == ESRCH)
158  {
159  _rev_thread_codes.erase(itr->second);
160  _thread_codes.erase(itr++); // post inc, takes copy before incr;
161  }
162  else
163  ++itr;
164  }
165 }

168 char Logger::get_thread_code(pthread_t tid)
169 {
170  tbb::mutex::scoped_lock guard(_mutex);
171 
172  ThreadCodes::const_iterator itr(_thread_codes.find(tid));
173  if (itr != _thread_codes.end())
174  return itr->second;
175 
176  for (char acode('A'); acode < 127; ++acode) // A-~ will allow for 86 threads
177  {
178  RevThreadCodes::const_iterator itr(_rev_thread_codes.find(acode));
179  if (itr == _rev_thread_codes.end())
180  {
181  _thread_codes.insert(ThreadCodes::value_type(tid, acode));
182  _rev_thread_codes.insert(ThreadCodes::value_type(acode, tid));
183  return acode;
184  }
185  }
186 
187  return '?';
188 }

202 bool FileLogger::rotate(bool force)
203 {
204  tbb::mutex::scoped_lock guard(_mutex);
205 
206  string thislFile(_pathname);
207 #ifdef HAVE_COMPRESSION
208  if (_flags & compress)
209  thislFile += ".gz";
210 #endif
211  if (_rotnum > 0 && (!_flags.has(append) || force))
212  {
213  vector<string> rlst(_rotnum);
214  rlst.push_back(thislFile);
215 
216  for (unsigned ii(0); ii < _rotnum && ii < max_rotation; ++ii)
217  {
218  ostringstream ostr;
219  ostr << _pathname << '.' << (ii + 1);
220  if (_flags & compress)
221  ostr << ".gz";
222  rlst.push_back(ostr.str());
223  }
224 
225  for (unsigned ii(_rotnum); ii; --ii)
226  rename (rlst[ii - 1].c_str(), rlst[ii].c_str()); // ignore errors
227  }
228 
229  delete _ofs;
230 
231  const ios_base::openmode mode (_flags & append ? ios_base::out | ios_base::app : ios_base::out);
232 #ifdef HAVE_COMPRESSION
233  if (_flags & compress)
234  _ofs = new ogzstream(thislFile.c_str(), mode);
235  else
236 #endif
237  _ofs = new ofstream(thislFile.c_str(), mode);
238 
239  return true;
240 }

244 {
245  const string pathname(fname.substr(1));

247  // | uses IFS safe cfpopen; ! uses old popen if available

249 #ifdef HAVE_POPEN

254 #ifdef DEBUG

261 #endif
262  ;
263  else
