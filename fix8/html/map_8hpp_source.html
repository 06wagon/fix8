885 template<typename T, typename kernelF, unsigned int blockSize, bool nIsPow2>
886 __global__ void reduceCUDAKernel(kernelF K, T *input, T *output, size_t sizen) {
887 
888  T *sdata = SharedMemory<T>();
889 
890  // perform first level of reduction,
891  // reading from global memory, writing to shared memory
892  unsigned int tid = threadIdx.x;
893  unsigned int i = blockIdx.x*blockSize*2 + threadIdx.x;
894  unsigned int gridSize = blockSize*2*gridDim.x;
895 
896  T result = 0;
897 
898  if(i < size) {
899  result = input[i];
900  // ensure we don't read out of bounds -- this is optimized away for powerOf2 sized arrays
901 
902  // There we pass it always false
903  if (nIsPow2 || i + blockSize < n)
904  result = K.K(result, input[i+blockSize]);
905  i += gridSize;
906  }
907 
908  // we reduce multiple elements per thread. The number is determined by the
909  // number of active thread blocks (via gridDim). More blocks will result
910  // in a larger gridSize and therefore fewer elements per thread
911  while(i < size) {
912  result = reduceFunc.CU(result, input[i]);
913  // ensure we don't read out of bounds -- this is optimized away for powerOf2 sized arrays
914  if (nIsPow2 || i + blockSize < n)
915  result = K.K(result, input[i+blockSize]);
916  i += gridSize;
917  }
918 
919  // each thread puts its local sum into shared memory
920  sdata[tid] = result;
921 
922  __syncthreads();
923 
924  // do reduction in shared mem
925  if (blockSize >= 512) { if (tid < 256) { sdata[tid] = result = K.K(result, sdata[tid + 256]); } __syncthreads(); }
926  if (blockSize >= 256) { if (tid < 128) { sdata[tid] = result = K.K(result, sdata[tid + 128]); } __syncthreads(); }
927  if (blockSize >= 128) { if (tid < 64) { sdata[tid] = result = K.K(result, sdata[tid + 64]); } __syncthreads(); }
928 
929  if (tid < 32) {
930  // now that we are using warp-synchronous programming (below)
931  // we need to declare our shared memory volatile so that the compiler
932  // doesn't reorder stores to it and induce incorrect behavior.
933  volatile T* smem = sdata;
934  if (blockSize >= 64) { smem[tid] = result = K.K(result, smem[tid + 32]); }
935  if (blockSize >= 32) { smem[tid] = result = K.K(result, smem[tid + 16]); }
936  if (blockSize >= 16) { smem[tid] = result = K.K(result, smem[tid + 8]); }
937  if (blockSize >= 8) { smem[tid] = result = K.K(result, smem[tid + 4]); }
938  if (blockSize >= 4) { smem[tid] = result = K.K(result, smem[tid + 2]); }
939  if (blockSize >= 2) { smem[tid] = result = K.K(result, smem[tid + 1]); }
940  }
941 
942  // write result for this block to global mem
943  if (tid == 0)
944  output[blockIdx.x] = sdata[0];
945 }
