95 unsigned MemcachedPersister::get(const unsigned from, const unsigned to, Session& session,
96  bool (Session::*callback)(const Session::SequencePair& with, Session::RetransmissionContext& rctx)) const
97 {
98  unsigned last_seq(0);
99  get_last_seqnum(last_seq);
100  unsigned recs_sent(0), startSeqNum(find_nearest_highest_seqnum (from, last_seq));
101  const unsigned finish(to == 0 ? last_seq : to);
102  Session::RetransmissionContext rctx(from, to, session.get_next_send_seq());
103 
104  if (!startSeqNum || from > finish)
105  {
106  GlobalLogger::log(FILE_LINE, "No records found");
107  rctx._no_more_records = true;
108  (session.*callback)(Session::SequencePair(0, ""), rctx);
109  return 0;
110  }
111 
112  for (; startSeqNum <= finish; ++startSeqNum)
113  {
114  string target, key(generate_seq_key(startSeqNum));
115  if (get_from_cache(key, target))
116  {
117  Session::SequencePair txresult(startSeqNum, f8String(target.c_str(), target.size()));
118  ++recs_sent;
119  if (!(session.*callback)(txresult, rctx))
120  break;
121  }
122  }
123 
124  rctx._no_more_records = true;
125  (session.*callback)(Session::SequencePair(0, ""), rctx);
126 
127  return recs_sent;
128 }
