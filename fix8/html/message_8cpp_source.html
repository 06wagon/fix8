51 unsigned MessageBase::extract_header(const f8String& from, char *len, char *mtype)
52 {
53  const char *dptr(from.data());
54  const unsigned flen(static_cast<unsigned>(from.size()));
55  char tag[MAX_MSGTYPE_FIELD_LEN], val[MAX_FLD_LENGTH];
56  unsigned s_offset(0), result;
57 
58  if ((result = extract_element(dptr, flen, tag, val)))
59  {
60  if (*tag != '8')
61  return 0;
62  s_offset += result;
63  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, len)))
64  {
65  if (*tag != '9')
66  return 0;
67  s_offset += result;
68  if ((result = extract_element(dptr + s_offset, flen - s_offset, tag, mtype)))
69  {
70  if (*tag != '3' || *(tag + 1) != '5')
71  return 0;
72  s_offset += result;
73  }
74  }
75  }
76  return s_offset;
77 }

87 unsigned MessageBase::decode(const f8String& from, unsigned s_offset, unsigned ignore, bool permissive_mode)
88 {
89  const unsigned fsize(static_cast<unsigned>(from.size()) - ignore), npos(0xffffffff);
90  unsigned pos(static_cast<unsigned>(_pos.size())), last_valid_pos(npos);
91  const char *dptr(from.data());
92  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
93  size_t last_valid_offset(0);
94 
95  for (unsigned result; s_offset <= fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
96  {
97  const unsigned short tv(fast_atoi<unsigned short>(tag));
98  Presence::const_iterator itr(_fp.get_presence().find(tv));
99  if (itr == _fp.get_presence().end())
100  {
101  if (permissive_mode)
102  {
103  if (last_valid_pos == npos)
104  {
105  last_valid_pos = pos;
106  last_valid_offset = s_offset;
107  }
108  s_offset += result;
109  continue;
110  }
111  break;
112  }
113  s_offset += result;
114  if (itr->_field_traits.has(FieldTrait::present))
115  {
116  if (!itr->_field_traits.has(FieldTrait::automatic))
117  throw DuplicateField(tv);
118  }
119  else
120  {
121  const BaseEntry *be(_ctx.find_be(tv));
122  if (!be)
123  throw UnknownField(tv);
124  BaseField *bf(be->_create._do(val, be->_rlm, -1));
125  add_field_decoder(tv, ++pos, bf);
126  itr->_field_traits.set(FieldTrait::present);
127  // check if repeating group and num elements > 0
128  if (itr->_field_traits.has(FieldTrait::group) && static_cast<Field<int, 0> *>(bf)->get() > 0)
129  s_offset = decode_group(tv, from, s_offset, ignore);
130  }
131  }
132 
133  const unsigned short missing(_fp.find_missing());
134  if (missing)
135  {
136  const BaseEntry *tbe(_ctx.find_be(missing));
137  ostringstream ostr;
138  ostr << tbe->_name << " (" << missing << ')';
139  throw MissingMandatoryField(ostr.str());
140  }
141 
142  return permissive_mode && last_valid_pos == pos ? static_cast<unsigned>(last_valid_offset) : s_offset;
143 }

146 unsigned MessageBase::decode_group(const unsigned short fnum, const f8String& from, unsigned s_offset, unsigned ignore)
147 {
148  unsigned result;
149  GroupBase *grpbase(find_group(fnum));
150  if (!grpbase)
151  throw InvalidRepeatingGroup(fnum);
152  const unsigned fsize(static_cast<unsigned>(from.size()) - ignore);
153  const char *dptr(from.data());
154  char tag[MAX_FLD_LENGTH], val[MAX_FLD_LENGTH];
155 
156  for (bool ok(true); ok && s_offset < fsize; )
157  {
158  unique_ptr<MessageBase> grp(grpbase->create_group());
159 
160  for (unsigned pos(0); s_offset < fsize && (result = extract_element(dptr + s_offset, fsize - s_offset, tag, val));)
161  {
162  const unsigned tv(fast_atoi<unsigned>(tag));
163  Presence::const_iterator itr(grp->_fp.get_presence().end());
164  if (grp->_fp.get(tv, itr, FieldTrait::present)) // already present; next group?
165  break;
166  if (pos == 0 && grp->_fp.getPos(tv, itr) != 1) // first field in group is mandatory
167  throw MissingRepeatingGroupField(tv);
168  const BaseEntry *be(_ctx.find_be(tv));
169  if (!be || !grp->_fp.has(tv, itr)) // unknown field or field not found in sub-group - end of repeats?
170  {
171  ok = false;
172  break;
173  }
174  s_offset += result;
175  BaseField *bf(be->_create._do(val, be->_rlm, -1));
176  grp->add_field(tv, itr, ++pos, bf, false);
177  grp->_fp.set(tv, itr, FieldTrait::present); // is present
178  // nested group (check if not zero elements)
179  if (grp->_fp.is_group(tv, itr) && static_cast<Field<int, 0> *>(bf)->get() > 0)
180  s_offset = grp->decode_group(tv, from, s_offset, ignore);
181  }
182 
183  const unsigned short missing(grp->_fp.find_missing());
184  if (missing)
185  {
186  const BaseEntry *tbe(_ctx.find_be(missing));
187  ostringstream ostr;
188  ostr << tbe->_name << " (" << missing << ')';
189  throw MissingMandatoryField(ostr.str());
190  }
191  *grpbase << grp.release();
192  }
193 
194  return s_offset;
195 }

204 Message *Message::factory(const F8MetaCntx& ctx, const f8String& from, bool no_chksum, bool permissive_mode)
205 {
206  char mtype[MAX_MSGTYPE_FIELD_LEN] {}, len[MAX_MSGTYPE_FIELD_LEN] {};
207  const unsigned hlen(extract_header(from, len, mtype));
208 
209  if (!hlen)
210  {
211  //cerr << "Message::factory throwing" << endl;
212  throw InvalidMessage(from);
213  }
214 
215  const unsigned mlen(fast_atoi<unsigned>(len));
216  const BaseMsgEntry *bme(ctx._bme.find_ptr(mtype));
217  if (!bme)
218  throw InvalidMessage(mtype);
219  Message *msg(bme->_create._do());
220 #if defined CODECTIMING
221  IntervalTimer itm;
222 #endif
223  msg->decode(from, hlen, 7, permissive_mode); // skip already decoded mandatory 8, 9, 35 and 10
224 #if defined CODECTIMING
225  _decode_timings._cpu_used += itm.Calculate().AsDouble();
226  ++_decode_timings._msg_count;
227 #endif
228 
229  msg->_header->get_body_length()->set(mlen);
230  msg->_header->get_msg_type()->set(mtype);
231 #if defined POPULATE_METADATA
232  msg->check_set_rlm(fitr->second);
233 #endif
234 
235  const char *pp(from.data() + from.size() - 7); // FIXME: assumes supplied string is one complete message only
236  if (*pp != '1' || *(pp + 1) != '0') // 10=XXX^A
237  throw InvalidMessage(from);
238  if (!no_chksum) // permit chksum calculation to be skipped
239  {
240  const f8String chksum(pp + 3, 3);
241  msg->_trailer->get_check_sum()->set(chksum);
242  const unsigned chkval(fast_atoi<unsigned>(chksum.c_str())), mchkval(calc_chksum(from, 0, static_cast<unsigned>(from.size()) - 7));
243  if (chkval != mchkval)
244  throw BadCheckSum(mchkval);
245  }
246 
247  return msg;
248 }

251 // copy all fields from this message to 'to' where the field is legal for 'to' and it is not
252 // already present in 'to'; includes repeating groups;
253 // if force, copy all fields regardless, replacing any existing, adding any new
254 unsigned MessageBase::copy_legal(MessageBase *to, bool force) const
255 {
256  unsigned copied{};
257  for (Presence::const_iterator itr(_fp.get_presence().begin()); itr != _fp.get_presence().end(); ++itr)
258  {
259  if (itr->_field_traits & FieldTrait::present && (force || (to->_fp.has(itr->_fnum) && !to->_fp.get(itr->_fnum))))
260  {
261  if (itr->_field_traits & FieldTrait::group)
262  {
263  GroupBase *gb(find_group(itr->_fnum)), *gb1(to->find_group(itr->_fnum));
264 
265  for (GroupElement::const_iterator gitr(gb->_msgs.begin()); gitr != gb->_msgs.end(); ++gitr)
266  {
267  MessageBase *grc(gb1->create_group());
268  copied += (*gitr)->copy_legal(grc, force);
269  *gb1 += grc;
270  }
271  }
272 
273  BaseField *nf(get_field(itr->_fnum)->copy());
274 #if defined POPULATE_METADATA
275  to->check_set_rlm(nf);
276 #endif
277  Presence::const_iterator fpitr(_fp.get_presence().end());
278  if (force && to->_fp.get(itr->_fnum, fpitr, FieldTrait::present))
279  delete to->replace(itr->_fnum, fpitr, nf);
280  else
281  to->add_field(nf);
282  ++copied;
283  }
284  }
285 
286  return copied;
287 }

290 size_t MessageBase::encode(char *to) const
291 {
292  const char *where(to);
293  for (Positions::const_iterator itr(_pos.begin()); itr != _pos.end(); ++itr)
294  {
295 #if defined POPULATE_METADATA
296  check_set_rlm(itr->second);
297 #endif
298  Presence::const_iterator fpitr(_fp.get_presence().end());
299  if (!_fp.get(itr->second->_fnum, fpitr, FieldTrait::suppress)) // some fields are not encoded until unsuppressed (eg. checksum)
300  {
301  to += itr->second->encode(to);
302  if (fpitr->_field_traits.has(FieldTrait::group))
303  to += encode_group(itr->second->_fnum, to);
304  }
305  }
306 
307  return to - where;
308 }

357 size_t Message::encode(char **hmsg_store) const
358 {
359  char *moffs(*hmsg_store + HEADER_CALC_OFFSET), *msg(moffs);
360 
361 #if defined CODECTIMING
362  IntervalTimer itm;
363 #endif
364 
365  if (!_header)
366  throw MissingMessageComponent("header");
367  _header->get_msg_type()->set(_msgType);
368 #if defined RAW_MSG_SUPPORT
369  msg += (_begin_payload = _header->encode(msg)); // start
370  msg += (_payload_len = MessageBase::encode(msg));
371 #else
372  msg += _header->encode(msg); // start
373  msg += MessageBase::encode(msg);
374 #endif
375  if (!_trailer)
376  throw MissingMessageComponent("trailer");
377  msg += _trailer->encode(msg);
378  const size_t msgLen(msg - moffs); // checksummable msglength
379  const size_t hlen(_ctx._preamble_sz +
380  (msgLen < 10 ? 1 : msgLen < 100 ? 2 : msgLen < 1000 ? 3 : msgLen < 10000 ? 4 :
381  msgLen < 100000 ? 5 : msgLen < 1000000 ? 6 : 7));
382  char *hmsg(moffs - hlen);
383  *hmsg_store = hmsg;
384 
385  if (!_header->get_begin_string())
386  throw MissingMandatoryField(Common_BeginString);
387  _header->_fp.clear(Common_BeginString, FieldTrait::suppress);
388  hmsg += _header->get_begin_string()->encode(hmsg);
389 
390  if (!_header->get_body_length())
391  throw MissingMandatoryField(Common_BodyLength);
392  _header->_fp.clear(Common_BodyLength, FieldTrait::suppress);
393 
394  _header->get_body_length()->set(static_cast<int>(msgLen));
395  hmsg += _header->get_body_length()->encode(hmsg);
396 
397  if (!_trailer->get_check_sum())
398  throw MissingMandatoryField(Common_CheckSum);
399  _trailer->get_check_sum()->set(fmt_chksum(calc_chksum(moffs - hlen, msgLen + hlen)));
400  _trailer->_fp.clear(Common_CheckSum, FieldTrait::suppress);
401  msg += _trailer->get_check_sum()->encode(msg);
402 
403 #if defined CODECTIMING
404  _encode_timings._cpu_used += itm.Calculate().AsDouble();
405  ++_encode_timings._msg_count;
406 #endif
407 
408  *msg = 0;
409 #if defined RAW_MSG_SUPPORT
410  _rawmsg.assign(*hmsg_store, msg - *hmsg_store);
411 #endif
412  return msg - *hmsg_store;
413 }
