609  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
610  {
611  enum { get_tag, get_value } state(get_tag);
612 
613  for (unsigned ii(0); ii < sz; ++ii)
614  {
615  switch (state)
616  {
617  case get_tag:
618  if (!isdigit(from[ii]))
619  {
620  if (from[ii] != '=')
621  {
622  *val = *tag = 0;
623  return 0;
624  }
625  state = get_value;
626  }
627  else
628  *tag++ = from[ii];
629  break;
630  case get_value:
631  if (from[ii] == default_field_separator)
632  {
633  *val = *tag = 0;
634  return ++ii;
635  }
636  *val++ = from[ii];
637  break;
638  }
639  }
640  *val = *tag = 0;
641  return 0;
642  }
