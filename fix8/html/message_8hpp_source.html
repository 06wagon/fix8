656  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
657  {
658  enum { get_tag, get_value } state(get_tag);
659 
660  for (unsigned ii(0); ii < sz; ++ii)
661  {
662  switch (state)
663  {
664  case get_tag:
665  if (!isdigit(from[ii]))
666  {
667  if (from[ii] != default_assignment_separator)
668  return *val = *tag = 0;
669  state = get_value;
670  }
671  else
672  *tag++ = from[ii];
673  break;
674  case get_value:
675  if (from[ii] == default_field_separator)
676  {
677  *val = *tag = 0;
678  return ++ii;
679  }
680  *val++ = from[ii];
681  break;
682  }
683  }
684  return *val = *tag = 0;
685  }
