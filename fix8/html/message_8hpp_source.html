860  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
861  {
862  enum { get_tag, get_value } state(get_tag);
863 
864  for (unsigned ii(0); ii < sz; ++ii)
865  {
866  switch (state)
867  {
868  case get_tag:
869  if (!isdigit(from[ii]))
870  {
871  if (from[ii] != default_assignment_separator)
872  return *val = *tag = 0;
873  state = get_value;
874  }
875  else
876  *tag++ = from[ii];
877  break;
878  case get_value:
879  if (from[ii] == default_field_separator)
880  {
881  *val = *tag = 0;
882  return ++ii;
883  }
884  *val++ = from[ii];
885  break;
886  }
887  }
888  return *val = *tag = 0;
889  }
