664  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
665  {
666  enum { get_tag, get_value } state(get_tag);
667 
668  for (unsigned ii(0); ii < sz; ++ii)
669  {
670  switch (state)
671  {
672  case get_tag:
673  if (!isdigit(from[ii]))
674  {
675  if (from[ii] != default_assignment_separator)
676  return *val = *tag = 0;
677  state = get_value;
678  }
679  else
680  *tag++ = from[ii];
681  break;
682  case get_value:
683  if (from[ii] == default_field_separator)
684  {
685  *val = *tag = 0;
686  return ++ii;
687  }
688  *val++ = from[ii];
689  break;
690  }
691  }
692  return *val = *tag = 0;
693  }
