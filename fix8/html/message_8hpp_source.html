883  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
884  {
885  enum { get_tag, get_value } state(get_tag);
886 
887  for (unsigned ii(0); ii < sz; ++ii)
888  {
889  switch (state)
890  {
891  case get_tag:
892  if (!isdigit(from[ii]))
893  {
894  if (from[ii] != default_assignment_separator)
895  return *val = *tag = 0;
896  state = get_value;
897  }
898  else
899  *tag++ = from[ii];
900  break;
901  case get_value:
902  if (from[ii] == default_field_separator)
903  {
904  *val = *tag = 0;
905  return ++ii;
906  }
907  *val++ = from[ii];
908  break;
909  }
910  }
