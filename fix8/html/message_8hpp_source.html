652  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
653  {
654  enum { get_tag, get_value } state(get_tag);
655 
656  for (unsigned ii(0); ii < sz; ++ii)
657  {
658  switch (state)
659  {
660  case get_tag:
661  if (!isdigit(from[ii]))
662  {
663  if (from[ii] != default_assignment_separator)
664  return *val = *tag = 0;
665  state = get_value;
666  }
667  else
668  *tag++ = from[ii];
669  break;
670  case get_value:
671  if (from[ii] == default_field_separator)
672  {
673  *val = *tag = 0;
674  return ++ii;
675  }
676  *val++ = from[ii];
677  break;
678  }
679  }
680  return *val = *tag = 0;
681  }
