877  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
878  {
879  enum { get_tag, get_value } state(get_tag);
880 
881  for (unsigned ii(0); ii < sz; ++ii)
882  {
883  switch (state)
884  {
885  case get_tag:
886  if (!isdigit(from[ii]))
887  {
888  if (from[ii] != default_assignment_separator)
889  return *val = *tag = 0;
890  state = get_value;
891  }
892  else
893  *tag++ = from[ii];
894  break;
895  case get_value:
896  if (from[ii] == default_field_separator)
897  {
898  *val = *tag = 0;
899  return ++ii;
900  }
901  *val++ = from[ii];
902  break;
903  }
904  }
905  return *val = *tag = 0;
906  }
