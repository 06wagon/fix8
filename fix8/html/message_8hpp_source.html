663  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
664  {
665  enum { get_tag, get_value } state(get_tag);
666 
667  for (unsigned ii(0); ii < sz; ++ii)
668  {
669  switch (state)
670  {
671  case get_tag:
672  if (!isdigit(from[ii]))
673  {
674  if (from[ii] != default_assignment_separator)
675  return *val = *tag = 0;
676  state = get_value;
677  }
678  else
679  *tag++ = from[ii];
680  break;
681  case get_value:
682  if (from[ii] == default_field_separator)
683  {
684  *val = *tag = 0;
685  return ++ii;
686  }
687  *val++ = from[ii];
688  break;
689  }
690  }
691  return *val = *tag = 0;
692  }
