886  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
887  {
888  enum { get_tag, get_value } state(get_tag);
889 
890  for (unsigned ii(0); ii < sz; ++ii)
891  {
892  switch (state)
893  {
894  case get_tag:
895  if (!isdigit(from[ii]))
896  {
897  if (from[ii] != default_assignment_separator)
898  return *val = *tag = 0;
899  state = get_value;
900  }
901  else
902  *tag++ = from[ii];
903  break;
904  case get_value:
905  if (from[ii] == default_field_separator)
906  {
907  *val = *tag = 0;
908  return ++ii;
909  }
910  *val++ = from[ii];
911  break;
912  }
913  }

915  }

1142  static unsigned calc_chksum(const char *from, const size_t sz, const unsigned offset=0, const int len=-1)
1143  {
1144 #if defined FIX8_SIZEOF_UNSIGNED_LONG && FIX8_SIZEOF_UNSIGNED_LONG == 8
1145  // steroid chksum algorithm by charles.cooper@lambda-tg.com
1146  // Basic strategy is to unroll the loop. normally adding in a loop
1147  // is slow because of the dependency, the cpu has to finish each loop
1148  // before it starts the next one and is unable to pipeline. one
1149  // way to get around this is to have multiple registers and have
1150  // multiple adds per loop, then consolidating at the end
1151  // as in unroll_checksum.
1152  // that is slow though because there is a lot of wasted space and
1153  // the cpu has to do a lot of adds.
1154  // key is that addition of a word is going to be faster
1155  // than addition of two chars, and also each word has 8 chars in it.
1156  // so we get 8 adds per loop. long1 + long2 is almost like
1157  // looping over two arrays of 8 chars and adding them element wise.
1158  // except for the overflow. so we have to keep track of the overflow
1159  // and get rid of it at the end.
1160 
1161  const unsigned long OVERFLOW_MASK (1UL << 8 | 1UL << 16 | 1UL << 24 | 1UL << 32 | 1UL << 40 | 1UL << 48 | 1UL << 56);
1162  unsigned long ret{}, overflow{};
1163  from += offset;
1164  const unsigned long elen (len != -1 ? len : sz);
1165  size_t ii{};
1166  for (; ii < (elen - elen % 8); ii += 8)
1167  {
1168  unsigned long expected_overflow((ret & OVERFLOW_MASK) ^ (OVERFLOW_MASK & *reinterpret_cast<const unsigned long*>(from + ii)));
1169  ret += *reinterpret_cast<const unsigned long*>(from + ii);
1170  overflow += (expected_overflow ^ ret) & OVERFLOW_MASK;
1171  }
1172  ret = COLLAPSE_INT64(ret);
1173  overflow = COLLAPSE_INT64(overflow);
1174  for (; ii < elen; ret += from[ii++]); // add up rest one by one
1175  return (ret - overflow) & 0xff;
1176 #else
1177  // native chksum algorithm
1178  unsigned val(0);
1179  const char *eptr(from + (len != -1 ? len + offset : sz - offset));
1180  for (const char *ptr(from + offset); ptr < eptr; val += *ptr++);
1181  return val % 256;
1182 #endif
1183  }
