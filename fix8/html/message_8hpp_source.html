709  static unsigned extract_element(const char *from, const unsigned sz, char *tag, char *val)
710  {
711  enum { get_tag, get_value } state(get_tag);
712 
713  for (unsigned ii(0); ii < sz; ++ii)
714  {
715  switch (state)
716  {
717  case get_tag:
718  if (!isdigit(from[ii]))
719  {
720  if (from[ii] != default_assignment_separator)
721  return *val = *tag = 0;
722  state = get_value;
723  }
724  else
725  *tag++ = from[ii];
726  break;
727  case get_value:
728  if (from[ii] == default_field_separator)
729  {
730  *val = *tag = 0;
731  return ++ii;
732  }
733  *val++ = from[ii];
734  break;
735  }
736  }
737  return *val = *tag = 0;
738  }
