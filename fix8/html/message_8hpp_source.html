253  static unsigned extract_element(const char *from, const unsigned sz, f8String& tag, f8String& val)
254  {
255  enum { get_tag, get_value } state(get_tag);
256  tag.clear();
257  val.clear();
258 
259  for (unsigned ii(0); ii < sz; ++ii)
260  {
261  switch (state)
262  {
263  case get_tag:
264  if (!isdigit(from[ii]))
265  {
266  if (from[ii] != '=')
267  return 0;
268  state = get_value;
269  }
270  else
271  tag += from[ii];
272  break;
273  case get_value:
274  if (from[ii] == default_field_separator)
275  return ++ii;
276  val += from[ii];
277  break;
278  }
279  }
280  return 0;
281  }
