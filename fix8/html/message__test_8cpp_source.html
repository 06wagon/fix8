307 TEST(message, calc_chksum)
308 {
309  f8String msg(
310  "8=FIX.4.2\0019=117\00135=5\00134=3725\001369=617\00152=20130304-07:25:37.403\001"
311  "49=CME\00150=G\00156=1G9125N\00157=ADMIN\001143=US,IL\00158=Logout confirmed.\001789=618\00110=121\001");
312 
313  EXPECT_EQ(unsigned(172), Message::calc_chksum(msg));

324  EXPECT_EQ(unsigned(16), Message::calc_chksum(msg.c_str(), msg.length(), 10, 6));
