67 size_t modp_dtoa(double value, char* str, int prec) // DD
68 {
69  /* Hacky test for NaN
70  * under -fast-math this won't work, but then you also won't
71  * have correct nan values anyways. The alternative is
72  * to link with libmath (bad) or hack IEEE double bits (bad)
73  */
74  if (! (value == value)) {
75  str[0] = 'n'; str[1] = 'a'; str[2] = 'n'; str[3] = '\0';
76  return 3; // DD
77  }
78  /* if input is larger than thres_max, revert to exponential */
79  const double thres_max = (double)(0x7FFFFFFF);
80 
81  double diff = 0.0;
82  char* wstr = str;
83 
84  if (prec < 0) {
85  prec = 0;
86  } else if (prec > 9) {
87  /* precision of >= 10 can lead to overflow errors */
88  prec = 9;
89  }
90 
91 
92  /* we'll work in positive values and deal with the
93  negative sign issue later */
94  int neg = 0;
95  if (value < 0) {
96  neg = 1;
97  value = -value;
98  }
99 
100 
101  int whole = (int) value;
102  double tmp = (value - whole) * pow10[prec];
103  uint32_t frac = (uint32_t)(tmp);
104  diff = tmp - frac;
105 
106  if (diff > 0.5) {
107  ++frac;
108  /* handle rollover, e.g. case 0.99 with prec 1 is 1.0 */
109  if (frac >= pow10[prec]) {
110  frac = 0;
111  ++whole;
112  }
113  } else if (diff == 0.5 && ((frac == 0) || (frac & 1))) {
114  /* if halfway, round up if odd, OR
115  if last digit is 0. That last part is strange */
116  ++frac;
117  }
118 
119  /* for very large numbers switch back to native sprintf for exponentials.
120  anyone want to write code to replace this? */
121  /*
122  normal printf behavior is to print EVERY whole number digit
123  which can be 100s of characters overflowing your buffers == bad
124  */
125  if (value > thres_max)
126  return sprintf(str, "%e", neg ? -value : value); // DD
127 
128  if (prec == 0) {
129  diff = value - whole;
130  if (diff > 0.5) {
131  /* greater than 0.5, round up, e.g. 1.6 -> 2 */
132  ++whole;
133  } else if (diff == 0.5 && (whole & 1)) {
134  /* exactly 0.5 and ODD, then round up */
135  /* 1.5 -> 2, but 2.5 -> 2 */
136  ++whole;
137  }
138  } else {
139  int count = prec;
140  // now do fractional part, as an unsigned number
141  do {
142  --count;
143  *wstr++ = (char)(48 + (frac % 10));
144  } while (frac /= 10);
145  // add extra 0s
146  while (count-- > 0) *wstr++ = '0';
147  // add decimal
148  *wstr++ = '.';
149  }
150 
151  // do whole part
152  // Take care of sign
153  // Conversion. Number is reversed.
154  do *wstr++ = (char)(48 + (whole % 10)); while (whole /= 10);
155  if (neg) {
156  *wstr++ = '-';
157  }
158  *wstr='\0';
159  strreverse(str, wstr-1);
160  return wstr - str; // DD
161 }
