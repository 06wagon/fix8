80 template<typename T>
81 class f8_scoped_lock_impl
82 {
83  T *_local_mutex;
84  bool _disabled;
85 
86  f8_scoped_lock_impl(const f8_scoped_lock_impl&);
87  f8_scoped_lock_impl& operator=(const f8_scoped_lock_impl&);
88 
89 public:
90  f8_scoped_lock_impl() : _local_mutex(), _disabled() {}
91  f8_scoped_lock_impl(T& mutex) : _disabled() { acquire(mutex); }
92  f8_scoped_lock_impl(T& mutex, bool disable) : _disabled(disable)
93  {
94  if (!_disabled)
95  acquire(mutex);
96  }
97  ~f8_scoped_lock_impl()
98  {
99  if (!_disabled && _local_mutex)
100  release();
101  }
102 
103  void acquire(T& mutex)
104  {
105  mutex.lock();
106  _local_mutex = &mutex;
107  }
108 
109  bool try_acquire(T& mutex)
110  {
111  bool result(mutex.try_lock());
112  if(result)
113  _local_mutex = &mutex;
114  return result;
115  }
116 
117  void release()
118  {
119  _local_mutex->unlock();
120  _local_mutex = 0;
121  }
122 };
