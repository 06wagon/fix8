174 {
175  int val;
176  bool server(false), reliable(false);
177  string clcf;
178  unsigned next_send(0), next_receive(0);
179 
180 #ifdef HAVE_GETOPT_LONG
181  option long_options[] =
182  {
183  { "help", 0, 0, 'h' },
184  { "version", 0, 0, 'v' },
185  { "log", 1, 0, 'l' },
186  { "config", 1, 0, 'c' },
187  { "server", 0, 0, 's' },
188  { "send", 1, 0, 'S' },
189  { "receive", 1, 0, 'R' },
190  { "quiet", 0, 0, 'q' },
191  { "reliable", 0, 0, 'r' },
192  { 0 },
193  };
194 
195  while ((val = getopt_long (argc, argv, GETARGLIST.c_str(), long_options, 0)) != -1)
196 #else
197  while ((val = getopt (argc, argv, GETARGLIST.c_str())) != -1)
198 #endif
199  {
200  switch (val)
201  {
202  case 'v':
203  cout << argv[0] << " for "PACKAGE" version "VERSION << endl;
204  cout << "Released under the GNU LESSER GENERAL PUBLIC LICENSE, Version 3. See <http://fsf.org/> for details." << endl;
205  return 0;
206  case ':': case '?': return 1;
207  case 'h': print_usage(); return 0;
208  case 'l': GlobalLogger::set_global_filename(optarg); break;
209  case 'c': clcf = optarg; break;
210  case 's': server = true; break;
211  case 'S': next_send = GetValue<unsigned>(optarg); break;
212  case 'R': next_receive = GetValue<unsigned>(optarg); break;
213  case 'q': quiet = true; break;
214  case 'r': reliable = true; break;
215  default: break;
216  }
217  }
218 
219  RandDev::init();
220 
221  signal(SIGTERM, sig_handler);
222  signal(SIGINT, sig_handler);
223  signal(SIGQUIT, sig_handler);
224 
225  try
226  {
227  const string conf_file(server ? clcf.empty() ? "myfix_server.xml" : clcf : clcf.empty() ? "myfix_client.xml" : clcf);
228 
229  if (server)
230  {
231  ServerSession<myfix_session_server>::Server_ptr
232  ms(new ServerSession<myfix_session_server>(TEX::ctx, conf_file, "TEX1"));
233 
234  for (unsigned scnt(0); !term_received; )
235  {
236  if (!ms->poll())
237  continue;
238  SessionInstance<myfix_session_server>::Instance_ptr
239  inst(new SessionInstance<myfix_session_server>(*ms));
240  if (!quiet)
241  inst->session_ptr()->control() |= Session::print;
242  ostringstream sostr;
243  sostr << "client(" << ++scnt << ") connection established.";
244  GlobalLogger::log(sostr.str());
245  inst->start(true, next_send, next_receive);
246  cout << "Session(" << scnt << ") finished." << endl;
247  inst->stop();
248  }
249  }
250  else
251  {
252  scoped_ptr<ClientSession<myfix_session_client> >
253  mc(reliable ? new ReliableClientSession<myfix_session_client>(TEX::ctx, conf_file, "DLD1")
254  : new ClientSession<myfix_session_client>(TEX::ctx, conf_file, "DLD1"));
255  if (!quiet)
256  mc->session_ptr()->control() |= Session::printnohb;
257 
258  if (!reliable)
259  {
260  const LoginParameters& lparam(mc->session_ptr()->get_login_parameters());
261  mc->start(false, next_send, next_receive, lparam._davi());
262  }
263  else
264  mc->start(false);
265 
266  MyMenu mymenu(*mc->session_ptr(), 0, cout);
267  char ch;
268  mymenu.get_tty().set_raw_mode();
269  while(!mymenu.get_istr().get(ch).bad() && !term_received && ch != 0x3 && mymenu.process(ch))
270  ;
271 
272  if (reliable)
273  {
274  // don't explicitly call mc->session_ptr()->stop() with reliable sessions
275  // before checking if the session is already shutdown - the framework will generally do this for you
276  if (!mc->session_ptr()->is_shutdown())
277  mc->session_ptr()->stop();
278  }
279  else
280  mc->session_ptr()->stop();
281 
282  mymenu.get_tty().unset_raw_mode();
283  }
284  }
285  catch (f8Exception& e)
286  {
287  cerr << "exception: " << e.what() << endl;
288  }
289  catch (exception& e) // also catches Poco::Net::NetException
290  {
291  cerr << "exception: " << e.what() << endl;
292  }
293 
294  if (term_received)
295  cout << "terminated." << endl;
296  return 0;
297 }
