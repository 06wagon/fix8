56  /* Usage example:
57  * // loop parallelization using 3 workers
58  * for(int i=0;i<N;++i) FF_PARFOR_BEGIN(for,i,0,N,1,1,3) {
59  * A[i]=f(i) ----> A[i]=f(i);
60  * } FF_PARFOR_END(for);
61  *
62  * parallel for + reduction:
63  *
64  * s=0; // loop parallelization using 3 workers
65  * for(int i=0;i<N;++i) FF_PARFOR_BEGIN(for,s,i,0,N,1,1,3) {
66  * s*=f(i) ----> s*=f(i);
67  * } FF_PARFOR_END(for,s,*);
68  *
69  *
70  * NOTE: inside the body of the PARFOR it is possible to use the
71  * 'ff_thread_id' const integer variable to identify the thread id
72  * running the sequential portion of the loop.
73  */

309 class forall_Scheduler: public ff_node {
310 protected:
311  std::vector<bool> active;
312  std::vector<std::pair<long,forall_task_t> > data;
313  std::vector<forall_task_t> taskv;
314 protected:
315  // initialize the data vector
316  virtual inline long init_data(long start, long stop) {
317  const long numtasks = std::ceil((stop-start)/(double)step);
318  long totalnumtasks = std::ceil(numtasks/(double)chunk);
319  long ntxw = totalnumtasks / nw;
320  long r = totalnumtasks % nw;
321  long tt = totalnumtasks;
322 
323  if (ntxw == 0 && r>1) {
324  ntxw = 1, r = 0;
325  }
326  data.resize(nw);
327  taskv.resize(8*nw); // 8 is the maximum n. of jumps, see the heuristic below
328 
329  long end, t=0, e;
330  for(int i=0;i<nw && totalnumtasks>0;++i, totalnumtasks-=t) {
331  t = ntxw + ((r>1 && (i<r))?1:0);
332  e = start + (t*chunk - 1)*step + 1;
333  end = (e<stop) ? e : stop;
334  data[i] = std::make_pair(t, forall_task_t(start,end));
335  start = (end-1)+step;
336  }
337 
338  if (totalnumtasks) {
339  assert(totalnumtasks==1);
340  data[nw-1].first += totalnumtasks;
341  data[nw-1].second.end = stop;
342  }
343 
344  // for(int i=0;i<nw;++i) {
345  // printf("W=%d %ld <%ld,%ld>\n", i, data[i].first, data[i].second.start, data[i].second.end);
346  // }
347  // printf("totaltasks=%ld\n", tt);
348 
349  return tt;
350  }
351 public:
352  forall_Scheduler(ff_loadbalancer* lb, long start, long stop, long step, long chunk, int nw):

358  totaltasks = init_data(0,0);
359  assert(totaltasks==0);

362  void* svc(void* t) {
363  if (t==NULL) {
364  bool skip1=false,skip2=false,skip3=false;
365  if (totaltasks==0) return NULL;
366  const long endchunk = (chunk-1)*step + 1;
367  int jump = 0;
368  moretask:
369  for(int wid=0;wid<nw;++wid) {
370  if (data[wid].first) {
371  long start = data[wid].second.start;
372  long end = std::min(start+endchunk, data[wid].second.end);
373  taskv[wid+jump].set(start, end);
374  lb->ff_send_out_to(&taskv[wid+jump], wid);
375 
376  --data[wid].first;
377  (data[wid].second).start = (end-1)+step;
378  active[wid]=true;
379  } else {
380  // if we do not have task at the beginning the thread is terminated
381  lb->ff_send_out_to(EOS, wid);
382  active[wid]=false;
383  skip1=skip2=skip3=true;
384  }
385  }
386  jump+=nw;
387  assert((jump / nw) <= 8);
388  // heuristic: try to assign more task at the very beginning
389  if (!skip1 && totaltasks>=10*nw) { skip1=true; goto moretask;}
390  if (!skip2 && totaltasks>=100*nw) { skip1=false; skip2=true; goto moretask;}
391  if (!skip3 && totaltasks>=1000*nw){ skip1=false; skip2=false; skip3=true; goto moretask;}
392 
393  return GO_ON;
394  }
395  if (--totaltasks <=0) return NULL;
396  auto task = (forall_task_t*)t;
397  const long endchunk = (chunk-1)*step + 1;
398  const int wid = lb->get_channel_id();
399  int id = wid;
400  for(int cnt=0;cnt<nw;++cnt) {
401  if (data[id].first) {
402  long start = data[id].second.start;
403  long end = std::min(start+endchunk, data[id].second.end);
404  task->set(start, end);
405  lb->ff_send_out_to(task, wid);
406  --data[id].first;
407  (data[id].second).start = (end-1)+step;
408  return GO_ON;
409  }
410  // no task available, trying to steal a task
411  id = (id+1) % nw;
412  }
413  if (active[wid]) {
414  lb->ff_send_out_to(EOS, wid); // the thread is terminated
415  active[wid]=false;
416  }
417  return GO_ON;
418  }
