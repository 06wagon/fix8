66 bool BDBPersister::initialise(const f8String& dbDir, const f8String& dbFname, bool purge)
67 {
68  if (_opened)
69  return true;
70 
71  _dbDir = dbDir;
72  _dbFname = dbFname;
73 
74  // Use concurrent db and default shared memory pool
75  _dbEnv.open(_dbDir.c_str(), DB_CREATE | DB_INIT_MPOOL | DB_INIT_CDB | DB_THREAD, 0);
76 
77  bool notFound(false);
78 
79  if (!purge)
80  {
81  try
82  {
83  _db->set_bt_compare(bt_compare_fcn);
84  _db->open(0, _dbFname.c_str(), 0, DB_BTREE, DB_THREAD, 0); // try and open existing if possible
85  unsigned last;
86  if (get_last_seqnum(last))
87  {
88  ostringstream ostr;
89  ostr << _dbFname << ": Last sequence is " << last;
90  GlobalLogger::log(FILE_LINE, ostr.str());
91  }
92  }
93  catch(DbException& dbe)
94  {
95  switch (dbe.get_errno())
96  {
97  case ENOENT:
98  case EACCES:
99  notFound = true;
100  break;
101  default:
102  {
103  ostringstream ostr;
104  ostr << "Error: opening existing database: " << dbe.what() << " (" << dbe.get_errno() << ')';
105  GlobalLogger::log(FILE_LINE, ostr.str());
106  }
107  return false;
108  }
109  }
110  }
111 
112  if (notFound || purge) // create a new one
113  {
114  try
115  {
116  _db->open(0, _dbFname.c_str(), 0, DB_BTREE, DB_CREATE | DB_THREAD, 0);
117  if (purge)
118  {
119  _db->truncate(0, 0, 0);
120  GlobalLogger::log(FILE_LINE, "Purged perist db");
121  }
122  }
123  catch(DbException& dbe)
124  {
125  ostringstream ostr;
126  ostr << "Error: creating new database: " << dbe.what() << " (" << dbe.get_errno() << ')';
127  GlobalLogger::log(FILE_LINE, ostr.str());
128  return false;
129  }
130 
131  _wasCreated = true;
132  }
133 
134  _thread.start();
135  return _opened = true;
136 
137 }

172 unsigned BDBPersister::get(const unsigned from, const unsigned to, Session& session,
173  bool (Session::*callback)(const Session::SequencePair& with, Session::RetransmissionContext& rctx)) const
174 {
175  unsigned last_seq(0);
176  get_last_seqnum(last_seq);
177  unsigned recs_sent(0), startSeqNum(find_nearest_highest_seqnum (from, last_seq));
178  const unsigned finish(to == 0 ? last_seq : to);
179  Session::RetransmissionContext rctx(from, to, session.get_next_send_seq());
180 
181  if (!startSeqNum || from > finish)
182  {
183  GlobalLogger::log(FILE_LINE, "No records found");
184  rctx._no_more_records = true;
185  (session.*callback)(Session::SequencePair(0, ""), rctx);
186  return 0;
187  }
188 
189  KeyDataBuffer buffer(startSeqNum);
190  KeyDataPair keyPair(buffer);
191  Dbc *cursorp;
192  _db->cursor (0, &cursorp, 0);
193  int retval;
194 
195  if ((retval = cursorp->get(&keyPair._key, &keyPair._data, DB_SET)) == 0)
196  {
197  do
198  {
199  const unsigned seqnum(buffer.keyBuf_.int_);
200  if (!seqnum || seqnum > finish)
201  break;
202  Session::SequencePair result(seqnum, buffer.dataBuf_);
203  ++recs_sent;
204  if (!(session.*callback)(result, rctx))
205  break;
206  }
207  while(cursorp->get(&keyPair._key, &keyPair._data, DB_NEXT) == 0);
208 
209  rctx._no_more_records = true;
210  (session.*callback)(Session::SequencePair(0, ""), rctx);
211  }
212  else
213  {
214  ostringstream ostr;
215  ostr << "record not found (" << db_strerror(retval) << ')';
216  GlobalLogger::log(FILE_LINE, ostr.str());
217  }
218  cursorp->close();
219 
220  return recs_sent;
221 }

299 int BDBPersister::operator()()
300 {
301  unsigned received(0), persisted(0);
302  bool stopping(false);
303 
304  for (;!_cancellation_token;)
305  {
306  KeyDataBuffer *msg_ptr(0);
307 
308 #if (MPMC_SYSTEM == MPMC_TBB)
309  KeyDataBuffer buffer;
310  if (stopping) // make sure we dequeue any pending msgs before exiting
311  {
312  if (!_persist_queue.try_pop(buffer))
313  break;
314  }
315  else
316  _persist_queue.pop (buffer); // will block
317  msg_ptr = &buffer;
318 
319  if (buffer.empty()) // means exit
320  {
321  stopping = true;
322  continue;
323  }
324 #else
325  _persist_queue.pop(msg_ptr); // will block
326  if (msg_ptr->empty()) // means exit
327  break;
328 #endif
329  //cout << "persisted..." << endl;
330 
331  ++received;
332 
333  if (msg_ptr)
334  {
335  KeyDataPair keyPair(*msg_ptr);
336  int retval(_db->put(0, &keyPair._key, &keyPair._data, 0)); // will overwrite if found
337  if (retval)
338  {
339  ostringstream ostr;
340  ostr << "Could not add" << '(' << db_strerror(retval) << ')';
341  GlobalLogger::log(FILE_LINE, ostr.str());
342  }
343  else
344  ++persisted;
345  }
346 #if (MPMC_SYSTEM == MPMC_FF)
347  _persist_queue.release(msg_ptr);
348 #endif
349  }
350 
351  //cout << "persister()() exited..." << endl;
352 
353  ostringstream ostr;
354  ostr << received << " messages received, " << persisted << " messages persisted";
355  GlobalLogger::log(FILE_LINE, ostr.str());
356 
357  return 0;
358 }
