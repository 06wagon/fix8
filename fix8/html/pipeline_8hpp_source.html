43  /* FF_PIPEA creates an n-stage accelerator pipeline (max 16 stages)
44  * where channels have type 'type'. The functions called in each stage
45  * must have the following signature:
46  * bool (*F)(type &)
47  * If the function F returns false an EOS is produced in the
48  * output channel.
49  *
50  * Usage example:
51  *
52  * struct task_t { .....};
53  *
54  * bool F(task_t &task) { .....; return true;}
55  * bool G(task_t &task) { .....; return true;}
56  * bool H(task_t &task) { .....; return true;}
57  *
58  * FF_PIPEA(pipe1, task_t, F, G); // 2-stage
59  * FF_PIPEA(pipe2, task_t, F, G, H); // 3-stage
60  *
61  * FF_PIPEARUN(pipe1); FF_PIPEARUN(pipe2);
62  * for(...) {
63  * if (even_task) FF_PIPEAOFFLOAD(pipe1, new task_t(...));
64  * else FF_PIPEAOFFLOAD(pipe2, new task_t(...));
65  * }
66  * FF_PIEAEND(pipe1); FF_PIPEAEND(pipe2);
67  * FF_PIEAWAIT(pipe1); FF_PIPEAWAIT(pipe2);
68  *
69  */
70 #define FF_PIPEA(pipename, type, ...) \
71  ff_pipeline _FF_node_##pipename(true); \
72  _FF_node_##pipename.cleanup_nodes(); \
73  { \
74  class _FF_pipe_stage: public ff_node { \
75  typedef bool (*type_f)(type &); \
76  type_f F; \
77  public: \
78  _FF_pipe_stage(type_f F):F(F) {} \
79  void* svc(void *task) { \
80  type* _ff_in = (type*)task; \
81  if (!F(*_ff_in)) return NULL; \
82  return task; \
83  } \
84  }; \
85  ff_pipeline *p = &_FF_node_##pipename; \
86  FOREACHPIPE(p->add_stage, __VA_ARGS__); \
87  }
88 
89 #define FF_PIPEARUN(pipename) _FF_node_##pipename.run()

91 #define FF_PIPEAOFFLOAD(pipename, task) _FF_node_##pipename.offload(task);

270  int nstages=static_cast<int>(nodes_list.size());

307  return 0;
