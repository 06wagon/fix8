75 INLINE static int usleep(unsigned long microsecs) {
76  if (microsecs > 100000)
77  /* At least 100 mS. Typical best resolution is ~ 15ms */
78  Sleep (microsecs/ 1000);
79  else
80  {
81  /* Use Sleep for the largest part, and busy-loop for the rest. */
82  static double frequency;
83  if (frequency == 0)
84  {
85  LARGE_INTEGER freq;
86  if (!QueryPerformanceFrequency (&freq))
87  {
88  /* Cannot use QueryPerformanceCounter. */
89  Sleep (microsecs / 1000);
90  return 0;
91  }
92  frequency = (double) freq.QuadPart / 1000000000.0;
93  }
94  long long expected_counter_difference = 1000 * microsecs * (long long) frequency;
95  int sleep_part = (int) (microsecs) / 1000 - 10;
96  LARGE_INTEGER before;
97  QueryPerformanceCounter (&before);
98  long long expected_counter = before.QuadPart +
99 expected_counter_difference;
100  if (sleep_part > 0)
101  Sleep (sleep_part);
102  for (;;)
103  {
104  LARGE_INTEGER after;
105  QueryPerformanceCounter (&after);
106  if (after.QuadPart >= expected_counter)
107  break;
108  }
109  }
110  return(0);
111 }

139 INLINE static int gettimeofday(struct timeval *tv, struct timezone *tz)
140 {
141  FILETIME ft;
142  unsigned __int64 tmpres = 0;
143  static int tzflag;
144 
145  if (NULL != tv)
146  {
147  GetSystemTimeAsFileTime(&ft);
148 
149  tmpres |= ft.dwHighDateTime;
150  tmpres <<= 32;
151  tmpres |= ft.dwLowDateTime;
152 
153  /*converting file time to unix epoch*/
154  tmpres -= DELTA_EPOCH_IN_MICROSECS;
155  tmpres /= 10; /*convert into microseconds*/
156  tv->tv_sec = (long)(tmpres / 1000000UL);
157  tv->tv_usec = (long)(tmpres % 1000000UL);
158  }
159 
160  if (NULL != tz)
161  {
162  if (!tzflag)
163  {
164  _tzset();
165  tzflag++;
166  }
167  tz->tz_minuteswest = _timezone / 60;
168  tz->tz_dsttime = _daylight;
169  }
170 
171  return 0;
172 }
