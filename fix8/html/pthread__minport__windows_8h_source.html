230 INLINE int pthread_cond_wait(pthread_cond_t RESTRICT * cv,
231  pthread_mutex_t RESTRICT * external_mutex) {
232  // Avoid race conditions.
233  EnterCriticalSection (&cv->waiters_count_lock_);
234 
235  // Increment count of waiters.
236  cv->waiters_count_++;
237 
238  // Store current generation in our activation record.
239  int my_generation = cv->wait_generation_count_;
240 
241  LeaveCriticalSection (&cv->waiters_count_lock_);
242  LeaveCriticalSection (external_mutex);
243 
244  for (;;) {
245  // Wait until the event is signaled.
246  WaitForSingleObject (cv->event_, INFINITE);
247 
248  EnterCriticalSection (&cv->waiters_count_lock_);
249  // Exit the loop when the <cv->event_> is signaled and
250  // there are still waiting threads from this <wait_generation>
251  // that haven't been released from this wait yet.
252  int wait_done = cv->release_count_ > 0
253  && cv->wait_generation_count_ != my_generation;
254  LeaveCriticalSection (&cv->waiters_count_lock_);
255 
256  if (wait_done)
257  break;
258  }
259 
260  EnterCriticalSection (external_mutex);
261  EnterCriticalSection (&cv->waiters_count_lock_);
262  cv->waiters_count_--;
263  cv->release_count_--;
264  int last_waiter = cv->release_count_ == 0;
265  LeaveCriticalSection (&cv->waiters_count_lock_);
266 
267  if (last_waiter)
268  // We're the last waiter to be notified, so reset the manual event.
269  ResetEvent (cv->event_);
270  return 0;
271 }

273 INLINE int pthread_cond_signal(pthread_cond_t *cv) {
274  EnterCriticalSection (&cv->waiters_count_lock_);
275  if (cv->waiters_count_ > cv->release_count_) {
276  SetEvent (cv->event_); // Signal the manual-reset event.
277  cv->release_count_++;
278  cv->wait_generation_count_++;
279  }
280  LeaveCriticalSection (&cv->waiters_count_lock_);
281  return 0;
282 }

284 INLINE int pthread_cond_broadcast(pthread_cond_t *cv) {
285  EnterCriticalSection (&cv->waiters_count_lock_);
286  if (cv->waiters_count_ > 0) {
287  SetEvent (cv->event_);
288  // Release all the threads in this generation.
289  cv->release_count_ = cv->waiters_count_;
290 
291  // Start a new generation.
292  cv->wait_generation_count_++;
293  }
294  LeaveCriticalSection (&cv->waiters_count_lock_);
295  return 0;
296 }
