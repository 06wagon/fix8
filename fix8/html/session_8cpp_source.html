251 bool Session::process(const f8String& from)
252 {
253  unsigned seqnum(0);
254 
255  try
256  {
257  const f8String::size_type fpos(from.find("34="));
258  if (fpos == f8String::npos)
259  {
260  slout_debug << "Session::process throwing for " << from;
261  throw InvalidMessage(from, FILE_LINE);
262  }
263 
264  seqnum = fast_atoi<unsigned>(from.data() + fpos + 3, default_field_separator);
265 
266  bool retry_plog(false);
267  if (_plogger && _plogger->has_flag(Logger::inbound))
268  {
269  if (_state != States::st_wait_for_logon)
270  plog(from, Logger::Info, 1);
271  else
272  retry_plog = true;
273  }
274 
275  const Message *msg(Message::factory(_ctx, from, _loginParameters._no_chksum_flag, _loginParameters._permissive_mode_flag));
276  if (!msg)
277  {
278  glout_fatal << "Fatal: factory failed to generate a valid message";
279  return false;
280  }
281 
282  if ((_control & printnohb) && msg->get_msgtype() != Common_MsgType_HEARTBEAT)
283  cout << *msg << endl;
284  else if (_control & print)
285  cout << *msg << endl;
286 
287  bool result(false), admin_result(msg->is_admin() ? handle_admin(seqnum, msg) : true);
288  if (msg->get_msgtype().size() > 1)
289  goto application_call;
290  else switch(msg->get_msgtype()[0])
291  {
292  default:
293 application_call:
294  if (activation_check(seqnum, msg))
295  result = handle_application(seqnum, msg);
296  break;
297  case Common_MsgByte_HEARTBEAT:
298  result = handle_heartbeat(seqnum, msg);
299  break;
300  case Common_MsgByte_TEST_REQUEST:
301  result = handle_test_request(seqnum, msg);
302  break;
303  case Common_MsgByte_RESEND_REQUEST:
304  result = handle_resend_request(seqnum, msg);
305  break;
306  case Common_MsgByte_REJECT:
307  result = handle_reject(seqnum, msg);
308  break;
309  case Common_MsgByte_SEQUENCE_RESET:
310  result = handle_sequence_reset(seqnum, msg);
311  break;
312  case Common_MsgByte_LOGOUT:
313  result = handle_logout(seqnum, msg);
314  break;
315  case Common_MsgByte_LOGON:
316  result = handle_logon(seqnum, msg);
317  break;
318  }
319 
320  ++_next_receive_seq;
321  if (retry_plog)
322  plog(from, Logger::Info, 1);
323  if (_persist)
324  {
325  f8_scoped_spin_lock guard(_per_spl, _connection->get_pmodel() == pm_coro);
326  _persist->put(_next_send_seq, _next_receive_seq);
327  //cout << "Persisted:" << _next_send_seq << " and " << _next_receive_seq << endl;
328  }
329  delete msg;
330  return result && admin_result;
331  }
332  catch (f8Exception& e)
333  {
334  slout_debug << "process:: f8exception" << ' ' << seqnum << ' ' << e.what();
335 
336  // log to both incase the exception was to do with the logger
337  slout_fatal << e.what();
338  glout_fatal << e.what();
339 
340  if (!e.force_logoff())
341  {
342  send(generate_reject(seqnum, e.what()));
343  }
344  else
345  {
346  if (_state == States::st_logon_received && !_loginParameters._silent_disconnect)
347  {
348  send(generate_logout(e.what()), true, 0, true); // so it won't increment
349  do_state_change(States::st_logoff_sent);
350  }
351  stop();
352  }
353  }
354  catch (Poco::Net::NetException& e)
355  {
356  slout_debug << "process:: Poco::Net::NetException";
357  slout_error << e.what();
358  }
359  catch (exception& e)
360  {
361  slout_debug << "process:: std::exception";
362  slout_error << e.what();
363  }
364 
365  return false;
366 }

416  throw BadSendingTime(ostr.str());
417  }
418  }
419

433 
434  if (_connection->get_role() == Connection::cn_initiator)

511

519  }
520 
521 #if 0

523  slout_info << *_schedule;

525  }
