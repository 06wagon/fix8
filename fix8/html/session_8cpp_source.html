166 int Session::start(Connection *connection, bool wait, const unsigned send_seqnum,
167  const unsigned recv_seqnum, const f8String davi)
168 {
169  if (_logger)
170  _logger->purge_thread_codes();
171  if (_plogger)
172  _plogger->purge_thread_codes();
173  _control.clear(shutdown);
174  log("Starting session");
175  _connection = connection; // takes owership
176  if (!_connection->connect()) // if already connected returns true
177  return -1;;
178  if (_connection->get_role() == Connection::cn_acceptor)
179  atomic_init(States::st_wait_for_logon); // important for server that this is done before connect
180  _connection->start();
181  log("Session connected");
182 
183  if (_connection->get_role() == Connection::cn_initiator)
184  {
185  atomic_init(States::st_not_logged_in);
186  if (_loginParamaters._reset_sequence_numbers)
187  _next_send_seq = _next_receive_seq = 1;
188  else
189  {
190  recover_seqnums();
191  if (send_seqnum)
192  _next_send_seq = send_seqnum;
193  if (recv_seqnum)
194  _next_receive_seq = recv_seqnum;
195  }
196 
197  Message *msg(generate_logon(_connection->get_hb_interval(), davi));
198  send(msg);
199  _state = States::st_logon_sent;
200  }
201  else
202  {
203  if (send_seqnum)
204  _req_next_send_seq = send_seqnum;
205  if (recv_seqnum)
206  _req_next_receive_seq = recv_seqnum;
207  }
208 
209  if (wait) // wait for
210  {
211  _connection->join();
212  }
213 
214  return 0;
215 }

251 bool Session::process(const f8String& from)
252 {
253  unsigned seqnum(0);
254 
255  try
256  {
257  const f8String::size_type fpos(from.find("34="));
258  if (fpos == f8String::npos)
259  {
260  //cerr << "Session::process throwing for " << from << endl;
261  throw InvalidMessage(from);
262  }
263 
264  seqnum = fast_atoi<unsigned>(from.data() + fpos + 3, default_field_separator);
265 
266  bool retry_plog(false);
267  if (_state != States::st_wait_for_logon)
268  plog(from, 1);
269  else
270  retry_plog = true;
271 
272  scoped_ptr<Message> msg(Message::factory(_ctx, from));
273  if (!msg.get())
274  {
275  GlobalLogger::log("Fatal: factory failed to generate a valid message");
276  return false;
277  }
278 
279  if (_control & print)
280  cout << *msg << endl;
281  bool result((msg->is_admin() ? handle_admin(seqnum, msg.get()) : true)
282  && (this->*_handlers.find_ref(msg->get_msgtype()))(seqnum, msg.get()));
283  ++_next_receive_seq;
284  if (retry_plog)
285  plog(from, 1);
286  if (_persist)
287  {
288  _persist->put(_next_send_seq, _next_receive_seq);
289  //cout << "Persisted:" << _next_send_seq << " and " << _next_receive_seq << endl;
290  }
291  return result;
292 
293  }
294  catch (f8Exception& e)
295  {
296  //cout << "process:: f8exception" << ' ' << seqnum << ' ' << e.what() << endl;
297 
298  log(e.what());
299  if (!e.force_logoff())
300  {
301  Message *msg(generate_reject(seqnum, e.what()));
302  send(msg);
303  }
304  else
305  {
306  if (_state != States::st_logon_received)
307  {
308  send(generate_logout(), 0, true); // so it won't increment
309  _state = States::st_logoff_sent;
310  }
311  stop();
312  }
313  }
314  catch (exception& e) // also catches Poco::Net::NetException
315  {
316  //cout << "process:: std::exception" << endl;
317 
318  log(e.what());
319  }
320 
321  return false;
322 }

338 bool Session::sequence_check(const unsigned seqnum, const Message *msg)
339 {
340  //cout << "seqnum:" << seqnum << " next_target_seq:" << _next_receive_seq
341  //<< " next_sender_seq:" << _next_send_seq << " state:" << _state << " next_receive_seq:" << _next_receive_seq << endl;
342 
343  if (seqnum > _next_receive_seq)
344  {
345  if (_state == States::st_resend_request_sent)
346  log("Resend request already sent");
347  if (_state == States::st_continuous)
348  {
349  Message *rmsg(generate_resend_request(_next_receive_seq));
350  _state = States::st_resend_request_sent;
351  send(rmsg);
352  }
353  else
354  throw InvalidMsgSequence(seqnum, _next_receive_seq);
355  return false;
356  }
357 
358  if (seqnum < _next_receive_seq)
359  {
360  poss_dup_flag pdf(false);
361  msg->Header()->get(pdf);
362  if (!pdf()) // poss dup not set so bad
363  throw MsgSequenceTooLow(seqnum, _next_receive_seq);
364  sending_time st;
365  msg->Header()->get(st);
366  orig_sending_time ost;
367  if (msg->Header()->get(ost) && ost() > st())
368  {
369  ostringstream ostr;
370  ost.print(ostr);
371  throw BadSendingTime(ostr.str());
372  }
373  }
374 
375  return true;
376 }

379 bool Session::handle_logon(const unsigned seqnum, const Message *msg)
380 {
381  _state = States::st_logon_received;
382 
383  if (_connection->get_role() == Connection::cn_initiator)
384  {
385  enforce(seqnum, msg);
386  heartbeat_interval hbi;
387  msg->get(hbi);
388  _connection->set_hb_interval(hbi());
389  _state = States::st_continuous;
390  ostringstream ostr;
391  ostr << "Client setting heartbeat interval to " << hbi();
392  log(ostr.str());
393  }
394  else // acceptor
395  {
396  default_appl_ver_id davi;
397  msg->get(davi);
398 
399  sender_comp_id sci;
400  msg->Header()->get(sci);
401  target_comp_id tci;
402  msg->Header()->get(tci);
403  SessionID id(_ctx._beginStr, tci(), sci());
404 
405  // important - these objects can't be created until we have a valid SessionID
406  if (!_logger)
407  _logger = _sf->create_logger(_sf->_ses, Configuration::session_log, &id);
408  if (!_plogger)
409  _plogger = _sf->create_logger(_sf->_ses, Configuration::protocol_log, &id);
410  if (!_persist)
411  _persist = _sf->create_persister(_sf->_ses, &id);
412 
413  if (_ctx.version() >= 4100 && msg->have(Common_ResetSeqNumFlag) && msg->get<reset_seqnum_flag>()->get())
414  {
415  log("Resetting sequence numbers");
416  _next_send_seq = _next_receive_seq = 1;
417  }
418  else
419  {
420  recover_seqnums();
421  if (_req_next_send_seq)
422  _next_send_seq = _req_next_send_seq;
423  if (_req_next_receive_seq)
424  _next_receive_seq = _req_next_receive_seq;
425  }
426 
427  if (authenticate(id, msg))
428  {
429  _sid = id;
430  enforce(seqnum, msg);
431  Message *msg(generate_logon(_connection->get_hb_interval(), davi()));
432  send(msg);
433  _state = States::st_continuous;
434  }
435  else
436  {
437  ostringstream ostr;
438  ostr << id << " failed authentication";
439  log(ostr.str());
440  stop();
441  _state = States::st_session_terminated;
442  return false;
443  }
444  }
445 
446  ostringstream ostr;
447  ostr << "Heartbeat interval is " << _connection->get_hb_interval();
448  log(ostr.str());
449 
450  _timer.schedule(_hb_processor, 1000); // check every second
451 
452  return true;
453 }

489 bool Session::handle_resend_request(const unsigned seqnum, const Message *msg)
490 {
491  enforce(seqnum, msg);
492 
493  if (_state != States::st_resend_request_received)
494  {
495  _state = States::st_resend_request_received;
496 
497  begin_seq_num begin;
498  end_seq_num end;
499

752 bool Session::send_process(Message *msg) // called from the connection (possibly on separate thread)
753 {
754  bool is_dup(msg->Header()->have(Common_PossDupFlag));
755 
756  if (!msg->Header()->have(Common_SenderCompID))
757  *msg->Header() += new sender_comp_id(_sid.get_senderCompID());
758  if (!msg->Header()->have(Common_TargetCompID))
759  *msg->Header() += new target_comp_id(_sid.get_targetCompID());
760 
761  if (msg->Header()->have(Common_MsgSeqNum))
762  {
763  if (!is_dup)
764  {
765  *msg->Header() += new poss_dup_flag(true);
766  is_dup = true;
767  }
768 
769  sending_time sendtime;
770  msg->Header()->get(sendtime);
771  *msg->Header() += new orig_sending_time(sendtime());
772  }
773  else
774  {
775  //cerr << "send_process: _next_send_seq = " << _next_send_seq << endl;
776  *msg->Header() += new msg_seq_num(msg->get_custom_seqnum() ? msg->get_custom_seqnum() : _next_send_seq);
777  }
778  *msg->Header() += new sending_time;
779 
780  try
781  {
782  //cout << "Sending:" << *msg;
783  modify_outbound(msg);
784  f8String output;
785  const unsigned enclen(msg->encode(output));
786  if (!_connection->send(output))
787  {
788  ostringstream ostr;
789  ostr << "Message write failed: " << enclen;
790  log(ostr.str());
791  return false;
792  }
793  _last_sent.now();
794  plog(output);
795 
796  //cout << "send_process" << endl;
797 
798  if (!is_dup)
799  {
800  if (_persist)
801  {
802  if (!msg->is_admin())
803  _persist->put(_next_send_seq, output);
804  _persist->put(_next_send_seq + 1, _next_receive_seq);
805  //cout << "Persisted (send):" << (_next_send_seq + 1) << " and " << _next_receive_seq << endl;
806  }
807 
808  if (!msg->get_custom_seqnum() && !msg->get_no_increment() && msg->get_msgtype() != Common_MsgType_SEQUENCE_RESET)
809  {
810  ++_next_send_seq;
811  //cout << "Seqnum now:" << _next_send_seq << " and " << _next_receive_seq << endl;
812  }
813  }
814  }
815  catch (f8Exception& e)
816  {
817  log(e.what());
818  }
819 
820  return true;
821 }
