212  bool test(bool prev=false)
213  {
214  Tickval now(true);
215  const Tickval today(now.get_ticks() - (now.get_ticks() % Tickval::day));
216  now.adjust(_toffset); // adjust for local utc offset
217  bool active(prev);
218 
219  if (_start_day < 0) // start/end day not specified; daily only
220  {
221  //cout >> now << ' ' >> (today + _start) << ' ' >> (today + _end) << endl;
222 
223  if (now.in_range(today + _start, today + _end))
224  {
225  if (!prev)
226  active = true;
227  }
228  else if (prev)
229  active = false;
230  }
231  else
232  {
233  struct tm result;
234  now.as_tm(result);
235 
236  //cout >> now << ' ' >> (today + _start) << ' ' >> (today + _end) << ' ' << result.tm_wday << endl;
237 
238  if (!prev)
239  {
240  if ( ((_start_day > _end_day && (result.tm_wday >= _start_day || result.tm_wday <= _end_day))
241  || (_start_day < _end_day && result.tm_wday >= _start_day && result.tm_wday <= _end_day))
242  && now.in_range(today + _start, today + _end))
243  active = true;
244  }
245  else if ( ((_start_day > _end_day && (result.tm_wday < _start_day && result.tm_wday > _end_day))
246  || (_start_day < _end_day && result.tm_wday >= _end_day))
247  && now > today + _end)
248  active = false;
249  }
250 
251  return active;
252  }
