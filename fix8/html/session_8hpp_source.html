243  bool test(bool prev=false)
244  {
245  Tickval now(true);
246  const Tickval today(now.get_ticks() - (now.get_ticks() % Tickval::day));
247  now.adjust(_toffset); // adjust for local utc offset
248  bool active(prev);
249 
250  if (_start_day < 0) // start/end day not specified; daily only
251  {
252  //cout >> now << ' ' >> (today + _start) << ' ' >> (today + _end) << endl;
253 
254  if (now.in_range(today + _start, today + _end))
255  {
256  if (!prev)
257  active = true;
258  }
259  else if (prev)
260  active = false;
261  }
262  else
263  {
264  struct tm result;
265  now.as_tm(result);
266 
267  //cout >> now << ' ' >> (today + _start) << ' ' >> (today + _end) << ' ' << result.tm_wday << endl;
268 
269  if (!prev)
270  {
271  if ( ((_start_day > _end_day && (result.tm_wday >= _start_day || result.tm_wday <= _end_day))
272  || (_start_day < _end_day && result.tm_wday >= _start_day && result.tm_wday <= _end_day))
273  && now.in_range(today + _start, today + _end))
274  active = true;
275  }
276  else if ( ((_start_day > _end_day && (result.tm_wday < _start_day && result.tm_wday > _end_day))
277  || (_start_day < _end_day && result.tm_wday >= _end_day))
278  && now > today + _end)
279  active = false;
280  }
281 
282  return active;
283  }
