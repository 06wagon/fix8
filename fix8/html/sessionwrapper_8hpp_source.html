439  int operator()()

453  }

457  this->_session->log(e.what());

462  }

465  //cout << "process:: std::exception" << endl;
466  this->_session->log(e.what());
467  }
468

470 
471  do // FIXME possible endless loop condition

473  ++current %= _servers.size();
474  }

477
