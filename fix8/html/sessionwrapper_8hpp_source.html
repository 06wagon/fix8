88 struct PocoSslContext
89 {
90  PocoSslContext(const SslContext& ctx, bool client)
91  {
92  if (ctx._valid)
93  {
94  Poco::SharedPtr<Poco::Net::PrivateKeyPassphraseHandler> phrase_handler(new Fix8PassPhraseHandler(!client));
95  Poco::SharedPtr<Poco::Net::InvalidCertificateHandler> cert_handler(new Fix8CertificateHandler(!client));
96  Poco::Net::initializeSSL();
97  _context = new Poco::Net::Context(
98  client ? Poco::Net::Context::CLIENT_USE : Poco::Net::Context::SERVER_USE,
99  ctx._private_key_file, ctx._certificate_file, ctx._ca_location,
100  static_cast<Poco::Net::Context::VerificationMode>(ctx._verification_mode), ctx._verification_depth,
101  ctx._load_default_cas, ctx._cipher_list);
102  if (client)
103  Poco::Net::SSLManager::instance().initializeClient(phrase_handler, cert_handler, _context);
104  else
105  Poco::Net::SSLManager::instance().initializeServer(phrase_handler, cert_handler, _context);
106  }
107  }
108 
109  ~PocoSslContext()
110  {
111  if (_context)
112  Poco::Net::uninitializeSSL();
113  }
114 
115  Poco::Net::Context::Ptr _context;
116  bool is_secure() const { return _context.get() != 0; }
117 };

400  this->_session->stop();
401  }
402  catch (f8Exception& e)
403  {
404  this->_session->log(e.what());
405  }
406  catch (Poco::Net::NetException& e)
407  {
408  this->_session->log(e.what());
409  }
410  catch (std::exception& e)
411  {
412  this->_session->log(e.what());
413  }
414 
415  delete this->_cc;
416  this->_cc = 0;
417  delete this->_sock;
418 
419  if (!excepted)
420  break;
421 
422  if (_failover_cnt)
423  {
424  ++_servers[_current]._retries;
425 
426  for (;;) // FIXME possible endless loop condition
427  {
428  if (_servers[_current]._max_retries && _servers[_current]._retries < _servers[_current]._max_retries)
429  break;
430  _servers[_current]._retries = 0; // reset for next time
431  ++_current %= _failover_cnt;
432  }
433  }
434 
435  hypersleep<h_milliseconds>(this->_loginParameters._login_retry_interval);
436  }
437 
438  _giving_up = true;
439 
440  return 0;
