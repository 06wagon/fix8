103  ctx._load_default_cas, ctx._cipher_list);

105  Poco::Net::SSLManager::instance().initializeClient(phrase_handler, cert_handler, _context);
106  else
107  Poco::Net::SSLManager::instance().initializeServer(phrase_handler, cert_handler, _context);
108  }
109  }

111  ~PocoSslContext()
112  {
113 #if POCO_VERSION >= 0x01040000
114  if (_context)
115  Poco::Net::uninitializeSSL();

345  int operator()()

347  while(!_cancellation_token)
348  {
349  ++_attempts;
350 
351  bool excepted(_failover_cnt == 0);
352  try
353  {
354  if (_failover_cnt)
355  {
356  std::ostringstream ostr;
357  ostr << "Trying " << _servers[_current]._hostname << '(' << (1 + _servers[_current]._retries) << "), "
358  << _attempts << " attempts so far";

371  this->_sock = new Poco::Net::StreamSocket;
372 #endif

375  this->_session->set_login_parameters(this->_loginParameters);
376  this->_session->set_session_config(this);
377  this->_session->start(this->_cc, true, _send_seqnum, _recv_seqnum, this->_loginParameters._davi());
378  _send_seqnum = _recv_seqnum = 0; // only set seqnums for the first time round
379  }
380  catch (Poco::TimeoutException& e)
381  {
382  this->_session->log(e.what(), Logger::Error);
383  excepted = true;
384  }
385  catch (f8Exception& e)
386  {
387  // std::cerr << e.what() << std::endl;
388  this->_session->log(e.what(), Logger::Error);
389  excepted = true;
390  }
391  catch (Poco::Net::InvalidAddressException& e)
392  {
393  this->_session->log(e.what(), Logger::Error);
394  excepted = true;
395  }
396 #if POCO_VERSION >= 0x01040000
397  catch (Poco::Net::InvalidSocketException& e)
398  {
399  this->_session->log(e.what(), Logger::Error);
400  excepted = true;
401  }
402 #endif
403  catch (Poco::Net::ServiceNotFoundException& e)
404  {
405  //std::cerr << e.what() << std::endl;
406  this->_session->log(e.what(), Logger::Error);
407  excepted = true;
408  }
409  catch (Poco::Net::ConnectionRefusedException& e)
410  {
411  this->_session->log(e.what(), Logger::Error);
412  excepted = true;
413  }
414  catch (Poco::Net::DNSException& e)
415  {
416  this->_session->log(e.what(), Logger::Error);
417  excepted = true;
418  }
419  catch (Poco::Net::InterfaceNotFoundException& e)
420  {
421  this->_session->log(e.what(), Logger::Error);
422  excepted = true;
423  }
424  catch (Poco::Net::NetException& e) // catch all other NetExceptions
425  {
426  this->_session->log(e.what(), Logger::Error);
427  excepted = true;
428  }
429  catch (std::exception& e)
430  {
431  //std::cout << "process:: std::exception" << endl;
432  this->_session->log(e.what(), Logger::Error);
433  excepted = true;
434  }
435 
436  //std::cout << "operator()():out of try" << std::endl;
437  try
438  {
439  this->_session->stop();
440  }
441  catch (f8Exception& e)
442  {
443  this->_session->log(e.what(), Logger::Error);
444  }
445  catch (Poco::Net::NetException& e)
446  {
447  this->_session->log(e.what(), Logger::Error);
448  }
449  catch (std::exception& e)
450  {
451  this->_session->log(e.what(), Logger::Error);
452  }
453 
454  delete this->_cc;
455  this->_cc = nullptr;
456  delete this->_sock;
457  this->_sock = nullptr;
458 
459  if (!excepted || (_failover_cnt == 0 && _attempts > this->_loginParameters._login_retries))
460  break;
461 
462  if (_failover_cnt)
463  {
464  ++_servers[_current]._retries;
465 
466  while (!_cancellation_token) // FIXME possible endless loop condition
467  {
468  if (_servers[_current]._max_retries && _servers[_current]._retries < _servers[_current]._max_retries)
469  break;
470  _servers[_current]._retries = 0; // reset for next time
471  ++_current %= _failover_cnt;
472  }
473  }
474 
475  hypersleep<h_milliseconds>(this->_loginParameters._login_retry_interval);
476  }
477 
478  _giving_up = true;
479 
480  return 0;
481  }
