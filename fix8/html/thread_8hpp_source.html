186  virtual int join(int timeoutInMs = 0)
187  {
188 #if (THREAD_SYSTEM == THREAD_PTHREAD)
189  void *rptr(&_exitval);
190  return pthread_join(_tid, &rptr) ? -1 : _exitval;
191 #elif (THREAD_SYSTEM == THREAD_POCO)
192  try
193  {
194  if (_thread.isRunning())
195  {
196  if (timeoutInMs == 0)
197  _thread.join();
198  else
199  _thread.join(timeoutInMs);
200  }
201  }
202  catch(const Poco::SystemException& ex)
203  {
204  // this is due to poco throws exceptions in case of thread was stopped already or not running
205  return -1;
206  }
207  catch(const Poco::TimeoutException& ex)
208  {
209  // this is due to poco throws exceptions in case of waiting for thread exit timed out
210  return -2;
211  }
212  return _exitval;
213 #elif (THREAD_SYSTEM == THREAD_TBB)
214  if ( _thread.get() && _thread->joinable() )
215  _thread->join();
216  return _exitval;
217 #endif
218  }
