323 #ifdef __APPLE__
324 // A simple spinlock that spins up to 100K times and then does a sched_yield to back-off.
325 // unlock() could just set _lock to false BUT that assumes that the spinlock was locked
326 // to begin with, which may not be the case. Therefore this implementation has the
327 // advantage of causing your thread to spin if you try to unlock something that is
328 // not locked, which I would think is a logic error that the caller should fix.
329 // The choice of 100K was arbitrary. The right way to set that parameter would be
330 // to keep track of how big x gets before thread starvation occurs and use that number.
331 // That number is going to be target- and use-case-dependent though.
332 class f8_spin_lock
333 {
334  bool _isLocked;
335 public:
336  f8_spin_lock() : _isLocked(false) {}
337  ~f8_spin_lock() {}
338 
339  void lock()
340  {
341  register int x = 0;
342  while(!__sync_bool_compare_and_swap(&_isLocked, false, true))
343  {
344  if(++x >= 100000)
345  {
346  x = 0;
347  sched_yield();
348  }
349  }
350  }
351  bool try_lock() { return _isLocked; }
352  void unlock()
353  {
354  register int x = 0;
355  while(!__sync_bool_compare_and_swap(&_isLocked, true, false))
356  {
357  if(++x >= 100000)
358  {
359  x = 0;
360  sched_yield();
361  }
362  }
363  }
364 };
365 #else

436  bool try_acquire(T& mutex)
437  {
438  bool result(mutex.try_lock());
439  if (result)

441  return result;
442  }
443 
444  void release()

448  _local_mutex->unlock();
