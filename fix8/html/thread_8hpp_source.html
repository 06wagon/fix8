145  virtual int join(int timeoutInMs = 0)
146  {
147 #if (THREAD_SYSTEM == THREAD_PTHREAD)
148  void *rptr(&_exitval);
149  return pthread_join(_tid, &rptr) ? -1 : _exitval;
150 #elif (THREAD_SYSTEM == THREAD_POCO)
151  try
152  {
153  if (_thread.isRunning())
154  {
155  if (timeoutInMs == 0)
156  _thread.join();
157  else
158  _thread.join(timeoutInMs);
159  }
160  }
161  catch(const Poco::SystemException& ex)
162  {
163  // this is due to poco throws exceptions in case of thread was stopped already or not running
164  return -1;
165  }
166  catch(const Poco::TimeoutException& ex)
167  {
168  // this is due to poco throws exceptions in case of waiting for thread exit timed out
169  return -2;
170  }
171  return _exitval;
172 #elif (THREAD_SYSTEM == THREAD_TBB)
173  if (_thread.get() && _thread->joinable())
174  _thread->join();
175  return _exitval;
176 #endif
177  }
