71 // Windows does not have clock_gettime, so we must supply it
72 inline int clock_gettime(int X, struct timespec *tv)
73 {
74  LARGE_INTEGER t;
75  FILETIME f;
76  double nanoseconds;
77  //static LARGE_INTEGER offset;
78  static double frequencyToNanoseconds;
79  static int initialized = 0;
80  static BOOL usePerformanceCounter = 0;
81 
82  if (!initialized)
83  {
84  LARGE_INTEGER performanceFrequency;
85  initialized = 1;
86  usePerformanceCounter = QueryPerformanceFrequency(&performanceFrequency);
87  if (usePerformanceCounter)
88  {
89  //QueryPerformanceCounter(&offset);
90  frequencyToNanoseconds = (double)performanceFrequency.QuadPart / 1000000000.;
91  }
92  else
93  {
94  //offset = getFILETIMEoffset();
95  frequencyToNanoseconds = 10.;
96  }
97  }
98  if (usePerformanceCounter)
99  QueryPerformanceCounter(&t);
100  else
101  {
102  GetSystemTimeAsFileTime(&f);
103  t.QuadPart = f.dwHighDateTime;
104  t.QuadPart <<= 32;
105  t.QuadPart |= f.dwLowDateTime;
106  }
107 
108  //t.QuadPart -= offset.QuadPart;
109  nanoseconds = (double)t.QuadPart / frequencyToNanoseconds;
110  t.QuadPart = nanoseconds;
111  tv->tv_sec = t.QuadPart / 1000000000;
112  tv->tv_nsec = t.QuadPart % 1000000000;
113  return 0;
114 }

122 inline void current_utc_time(struct timespec *ts)
123 {
124 #ifdef __APPLE__
125  clock_serv_t cclock;
126  mach_timespec_t mts;
127  host_get_clock_service(mach_host_self(), CALENDAR_CLOCK, &cclock);
128  clock_get_time(cclock, &mts);
129  mach_port_deallocate(mach_task_self(), cclock);
130  ts->tv_sec = mts.tv_sec;
131  ts->tv_nsec = mts.tv_nsec;
132 #else
133  clock_gettime(CLOCK_REALTIME, ts);
134 #endif
135 }
