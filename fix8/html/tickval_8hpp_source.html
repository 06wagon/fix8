68 inline int clock_gettime(int X, struct timespec *tv)
69 {
70  LARGE_INTEGER t;
71  FILETIME f;
72  double nanoseconds;
73  //static LARGE_INTEGER offset;
74  static double frequencyToNanoseconds;
75  static int initialized = 0;
76  static BOOL usePerformanceCounter = 0;
77 
78  if (!initialized) {
79  LARGE_INTEGER performanceFrequency;
80  initialized = 1;
81  usePerformanceCounter = QueryPerformanceFrequency(&performanceFrequency);
82  if (usePerformanceCounter) {
83  //QueryPerformanceCounter(&offset);
84  frequencyToNanoseconds = (double)performanceFrequency.QuadPart / 1000000000.;
85  } else {
86  //offset = getFILETIMEoffset();
87  frequencyToNanoseconds = 10.;
88  }
89  }
90  if (usePerformanceCounter) QueryPerformanceCounter(&t);
91  else {
92  GetSystemTimeAsFileTime(&f);
93  t.QuadPart = f.dwHighDateTime;
94  t.QuadPart <<= 32;
95  t.QuadPart |= f.dwLowDateTime;
96  }
97 
98  //t.QuadPart -= offset.QuadPart;
99  nanoseconds = (double)t.QuadPart / frequencyToNanoseconds;
100  t.QuadPart = nanoseconds;
101  tv->tv_sec = t.QuadPart / 1000000000;
102  tv->tv_nsec = t.QuadPart % 1000000000;
103  return (0);
104 }
