142 template<typename T>
143 int Timer<T>::operator()()
144 {
145  unsigned elapsed(0);
146 
147  while(!_cancellation_token)
148  {
149  bool shouldsleep(false);
150  {
151  f8_scoped_spin_lock guard(_spin_lock);
152 
153  if (_event_queue.size())
154  {
155  TimerEvent<T> op(_event_queue.top());
156  if (!op._t) // empty timeval means exit
157  {
158  _event_queue.pop(); // remove from queue
159  break;
160  }
161 
162  const Tickval now(Tickval::get_tickval());
163  if (op._t <= now) // has elapsed
164  {
165  TimerEvent<T> rop(_event_queue.top()); // take a copy
166  _event_queue.pop(); // remove from queue
167  guard.release();
168  ++elapsed;
169  if (!(_monitor.*rop._callback)())
170  break;
171  if (op._repeat)
172  {
173  op._t = now.get_ticks() + op._intervalMS * Tickval::million;
174  guard.acquire(_spin_lock);
175  _event_queue.push(op); // push back on queue
176  }
177  }
178  else
179  shouldsleep = true;
180  }
181  else
182  shouldsleep = true;
183  } // we want the lock to go out of scope before we sleep
184 
185  if (shouldsleep)
186  hypersleep<h_milliseconds>(_granularity);
187  }
188 
189  std::ostringstream ostr;
190  ostr << "Terminating Timer thread (" << elapsed << " elapsed, " << _event_queue.size() << " queued).";
191  GlobalLogger::instance()->send(ostr.str());
192  return 0;
193 }

212 template<typename T>
213 bool Timer<T>::schedule(TimerEvent<T> what, unsigned timeToWait)
214 {
215  Tickval tofire;
216 
217  if (timeToWait)
218  {
219  // Calculate time to fire
220  Tickval::get_tickval(tofire);
221  tofire += timeToWait * Tickval::million;
222  what._intervalMS = timeToWait;
223  }
224 
225  what.set(tofire);
226  f8_scoped_spin_lock guard(_spin_lock);
227  _event_queue.push(what);
228 
229  return true;
230 }
