130 template<typename T>
131 int Timer<T>::operator()()
132 {
133  const struct timespec tspec = { 0, 1000 * 100 * _granularity }; // if _granularity == 1 : 0.1ms at 1000Hz; 1ms at 100Hz
134  unsigned elapsed(0);
135 
136  while(true)
137  {
138  bool shouldsleep(false);
139  {
140  tbb::mutex::scoped_lock guard(_mutex);
141 
142  if (_event_queue.size())
143  {
144  const TimerEvent<T>& op(_event_queue.top());
145  if (!op._t) // empty timeval means exit
146  {
147  _event_queue.pop(); // remove from queue
148  break;
149  }
150 
151  if (op._t < Tickval::get_tickval()) // has elapsed
152  {
153  const TimerEvent<T> rop(_event_queue.top()); // take a copy
154  _event_queue.pop(); // remove from queue
155  guard.release();
156  ++elapsed;
157  if (!(_monitor.*rop._callback)())
158  break;
159  }
160  else
161  shouldsleep = true;
162  }
163  else
164  shouldsleep = true;
165  } // we want the lock to go out of scope before we sleep
166 
167  if (shouldsleep)
168  rnanosleep(tspec);
169  }
170 
171  std::ostringstream ostr;
172  ostr << "Terminating Timer thread (" << elapsed << " elapsed, " << _event_queue.size() << " queued).";
173  GlobalLogger::instance()->send(ostr.str());
174  return 0;
175 }

194 template<typename T>
195 bool Timer<T>::schedule(const TimerEvent<T>& what, const unsigned timeToWait)
196 {
197  Tickval tofire;
198 
199  if (timeToWait)
200  {
201  // Calculate time to fire
202  Tickval::get_tickval(tofire);
203  tofire += timeToWait * Tickval::million;
204  }
205 
206  what.set(tofire);
207  tbb::mutex::scoped_lock guard(_mutex);
208  _event_queue.push(what);
209 
210  return true;
211 }
