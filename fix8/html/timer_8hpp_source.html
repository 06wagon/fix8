125 template<typename T>
126 int Timer<T>::operator()()
127 {
128  const struct timespec tspec = { 0, 1000 * 100 * _granularity }; // if _granularity == 1 : 0.1ms at 1000Hz; 1ms at 100Hz
129  unsigned elapsed(0);
130 
131  while(true)
132  {
133  bool shouldsleep(false);
134  {
135  tbb::mutex::scoped_lock guard(_mutex);
136 
137  if (_event_queue.size())
138  {
139  const TimerEvent<T>& op(_event_queue.top());
140  if (!op._t) // empty timeval means exit
141  {
142  _event_queue.pop(); // remove from queue
143  break;
144  }
145 
146  if (op._t < Tickval::get_tickval()) // has elapsed
147  {
148  const TimerEvent<T> rop(_event_queue.top()); // take a copy
149  _event_queue.pop(); // remove from queue
150  guard.release();
151  ++elapsed;
152  if (!(_monitor.*rop._callback)())
153  break;
154  }
155  else
156  shouldsleep = true;
157  }
158  else
159  shouldsleep = true;
160  } // we want the lock to go out of scope before we sleep
161 
162  if (shouldsleep)
163  rnanosleep(tspec);
164  }
165 
166  std::ostringstream ostr;
167  ostr << "Terminating Timer thread (" << elapsed << " elapsed, " << _event_queue.size() << " queued).";
168  GlobalLogger::instance()->send(ostr.str());
169  return 0;
170 }

173 template<typename T>
174 bool Timer<T>::schedule(const TimerEvent<T>& what, const unsigned timeToWait, const bool hi_res)
175 {
176  Tickval tofire;
177 
178  if (timeToWait)
179  {
180  // Calculate time to fire; hi_res = usecs, lo_res = msecs
181  Tickval::get_tickval(tofire);
182  tofire += timeToWait * (hi_res ? Tickval::thousand : Tickval::million);
183  }
184 
185  what.set(tofire);
186  tbb::mutex::scoped_lock guard(_mutex);
187  _event_queue.push(what);
188 
189  return true;
190 }
