341  result insert(const_iterator what)
342  {
343  if (!_sz)
344  {
345  _arr = new FieldTrait[_rsz];
346  memcpy(_arr, what, sizeof(FieldTrait));
347  ++_sz;

353  if (answer) // sorry already here
354  return std::make_pair(end(), false);
355 
356  if (_sz < _rsz) // we have space
357  {
358  memmove(where + 1, where, (end() - where) * sizeof(FieldTrait));
359  memcpy(where, what, sizeof(FieldTrait));
360  }
361  else // we have to make space
362  {
363  iterator new_arr(new FieldTrait[_rsz = _sz + calc_reserve(_sz, _reserve)]);
364  const size_t wptr(where - _arr);
365  if (wptr > 0)
366  memcpy(new_arr, _arr, sizeof(FieldTrait) * wptr);
367  memcpy(new_arr + wptr, what, sizeof(FieldTrait));
368  memcpy(new_arr + wptr + 1, where, (end() - where) * sizeof(FieldTrait));
369  delete[] _arr;
370  _arr = new_arr;
371  }
372  ++_sz;
373  return std::make_pair(where, true);
