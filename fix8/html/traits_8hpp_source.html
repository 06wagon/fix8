310  result insert(const_iterator what)
311  {
312  if (!_sz)

336  memcpy(new_arr + wptr, what, sizeof(FieldTrait));
337  memcpy(new_arr + wptr + 1, where, (end() - where) * sizeof(FieldTrait));
