209 static inline double ffTime(int tag, bool lock=false) {
210  static struct timeval tv_start = {0,0};
211  static struct timeval tv_stop = {0,0};
212  // needed to protect gettimeofday
213  // if multiple threads call ffTime
214  static lock_t L = {0};
215 
216  double res=0.0;
217  switch(tag) {
218  case START_TIME:{
219  spin_lock(L);
220  gettimeofday(&tv_start,NULL);
221  spin_unlock(L);
222  } break;
223  case STOP_TIME:{
224  spin_lock(L);
225  gettimeofday(&tv_stop,NULL);
226  spin_unlock(L);
227  res = diffmsec(tv_stop,tv_start);
228  } break;
229  case GET_TIME: {
230  res = diffmsec(tv_stop,tv_start);
231  } break;
232  default:
233  res=0;
234  }
235  return res;
236 }
